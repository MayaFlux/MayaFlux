13/**
14 * @class ImageCVProcessor
15 * @brief BufferProcessor executing a Kinesis::Vision pipeline on a GpuImageSource buffer.
16 *
17 * Renamed from ImageCVProcessor. ImageCVProcessor makes the domain
18 * (computer vision on image data) explicit and avoids collision with
19 * VisionProcessor (the Kakshya DataProcessor over pixel containers).
20 *
21 * m_gpu_staging is a persistent host-visible VKBuffer allocated once in
22 * on_attach, sized to the image footprint. Passed to download_and_normalise
23 * on every call so TextureLoom::download_data uses the fenced path rather
24 * than waitIdle, avoiding graphics queue stalls on each frame download.
25 *
26 * @tparam T A VKBuffer subclass satisfying GpuImageSource.
27 */
28template <GpuImageSource T>
29class ImageCVProcessor : public BufferProcessor {
30public:
31 /**
32 * @brief Construct with the vision pipeline to execute each processing_function call.
33 * @param sequence Ordered VisionSteps describing the pipeline.

38 m_processing_token = ProcessingToken::GRAPHICS_BACKEND;
39 }

43 /**
44 * @brief Validate the buffer type and reset executor state.
45 *
46 * Throws std::invalid_argument if the buffer cannot be cast to T.
47 * Called automatically by BufferProcessingChain::add_processor.
48 *
49 * @param buffer The Buffer to attach to.
50 */
51 void on_attach(const std::shared_ptr<Buffer>& buffer) override
52 {
53 auto typed = std::dynamic_pointer_cast<T>(buffer);
54 if (!typed) {
55 error<std::invalid_argument>(

73 /**
74 * @brief Clear state and reset executor.
75 * @param buffer The Buffer being detached.

84 /**
85 * @brief Download the current GPU image, run the VisionSequence, store the result.
86 *
87 * No-op if the buffer has expired or the image is unavailable.
88 *
89 * @param buffer The GpuImageSource buffer to read from.
90 */
91 void processing_function(const std::shared_ptr<Buffer>& /*buffer*/) override
92 {
93 auto typed = m_buffer.lock();
94 if (!typed)
95 return;
96
97 auto image = resolve_gpu_image(*typed);
98 if (!image || !image->is_initialized())
99 return;
100
101 auto frame = download_and_normalise(image, m_raw_staging, m_float_work, m_gpu_staging);
102 if (frame.empty())
103 return;
104
105 m_is_processing.store(true, std::memory_order_release);
106 m_result = m_executor.run(
107 m_sequence, frame,
108 image->get_width(), image->get_height());
109
110 if (m_result_source)
111 m_result_source->signal(m_result);
112
113 m_is_processing.store(false, std::memory_order_release);
114 }
115
116 /**
117 * @brief Replace the pipeline and reset inter-frame executor state.
118 *
119 * Not thread-safe relative to processing_function. Call only when idle.
120 *
121 * @param sequence Replacement VisionSequence.
122 */
123 void set_sequence(Kinesis::Vision::VisionSequence sequence)
124 {
125 m_sequence = std::move(sequence);
126 m_executor.reset();
127 }

131 *

134 * @return Most recent VisionResult.

149 {
150 if (!m_result_source) {
151 m_result_source = std::make_shared<
152 Vruta::BroadcastSource<Kinesis::Vision::VisionResult>>();
153 }
154 return m_result_source;
155 }
156
