27void DescriptorBindingsProcessor::bind_scalar_node(
28 const std::string& name,
29 const std::shared_ptr<Nodes::Node>& node,
30 const std::string& descriptor_name,
31 uint32_t set,
32 vk::DescriptorType type,
33 ProcessingMode mode)
34{
35 if (!node) {
36 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
37 "Cannot bind null node '{}'", name);
38 return;
39 }
40
41 if (m_config.bindings.find(descriptor_name) == m_config.bindings.end()) {
42 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
43 "Descriptor '{}' not found in shader config", descriptor_name);
44 return;
45 }
46
47 const auto& binding_config = m_config.bindings[descriptor_name];
48
49 auto gpu_buffer = create_descriptor_buffer(sizeof(float), type);
50
51 auto [it, inserted] = m_bindings.try_emplace(name);
52 auto& binding = it->second;
53
54 binding.node = node;
55 binding.descriptor_name = descriptor_name;
56 binding.set_index = set;
57 binding.binding_index = binding_config.binding;
58 binding.type = type;
59 binding.binding_type = BindingType::SCALAR;
60 binding.gpu_buffer = gpu_buffer;
61 binding.buffer_offset = 0;
62 binding.buffer_size = sizeof(float);
63 binding.processing_mode.store(mode, std::memory_order_release);
64
65 bind_buffer(descriptor_name, gpu_buffer);
66
67 m_needs_descriptor_rebuild = true;
68
69 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
70 "Bound scalar node '{}' to descriptor '{}' (mode: {})",
71 name, descriptor_name, mode == ProcessingMode::INTERNAL ? "INTERNAL" : "EXTERNAL");

274void DescriptorBindingsProcessor::execute_shader(const std::shared_ptr<VKBuffer>& buffer)
275{
276 auto& bindings_list = buffer->get_pipeline_context().descriptor_buffer_bindings;
277
278 for (auto& [name, binding] : m_bindings) {
279 update_descriptor_from_node(binding);
280
281 bool found = false;
282 for (auto& existing_binding : bindings_list) {
283 if (existing_binding.set == binding.set_index && existing_binding.binding == binding.binding_index) {
284 existing_binding.buffer_info.buffer = binding.gpu_buffer->get_buffer();
285 existing_binding.buffer_info.offset = binding.buffer_offset;
286 existing_binding.buffer_info.range = binding.buffer_size;
287 found = true;
288 break;
289 }
290 }
291
292 if (!found) {
293 bindings_list.push_back({ .set = binding.set_index,
294 .binding = binding.binding_index,
295 .type = binding.type,
296 .buffer_info = vk::DescriptorBufferInfo {
297 binding.gpu_buffer->get_buffer(),
298 binding.buffer_offset,
299 binding.buffer_size } });
300 }
301 }

317void DescriptorBindingsProcessor::update_descriptor_from_node(DescriptorBinding& binding)
318{
319 if (!binding.node) {
320 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
321 "Binding has null node");
322 return;
323 }
324
325 float value {};
326
327 if (binding.processing_mode.load(std::memory_order_acquire) == ProcessingMode::INTERNAL) {
328 value = static_cast<float>(Buffers::extract_single_sample(binding.node));
329 } else {
330 value = static_cast<float>(binding.node->get_last_output());
331 }
332 Nodes::NodeContext& ctx = binding.node->get_last_context();
333
334 switch (binding.binding_type) {
335 case BindingType::SCALAR: {
336 ensure_buffer_capacity(binding, sizeof(float));
337 upload_to_gpu(&value, sizeof(float), binding.gpu_buffer, nullptr);
338 break;
339 }
340
341 case BindingType::VECTOR: {
342 auto* gpu_vec = dynamic_cast<Nodes::GpuVectorData*>(&ctx);
343 if (!gpu_vec || !gpu_vec->has_gpu_data()) {
344 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
345 "Node context does not provide GpuVectorData");
346 return;
347 }
348
349 auto data = gpu_vec->gpu_data();
350
351 ensure_buffer_capacity(binding, data.size_bytes());
352 upload_to_gpu(data.data(), data.size_bytes(), binding.gpu_buffer, nullptr);
353 break;
354 }
355
356 case BindingType::MATRIX: {
357 auto* gpu_mat = dynamic_cast<Nodes::GpuMatrixData*>(&ctx);
358 if (!gpu_mat || !gpu_mat->has_gpu_data()) {
359 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
360 "Node context does not provide GpuMatrixData");
361 return;
362 }
363
364 auto data = gpu_mat->gpu_data();
365 ensure_buffer_capacity(binding, data.size_bytes());
366 upload_to_gpu(data.data(), data.size_bytes(), binding.gpu_buffer, nullptr);
367 break;
368 }
369
370 case BindingType::STRUCTURED: {
371 auto* gpu_struct = dynamic_cast<Nodes::GpuStructuredData*>(&ctx);
372 if (!gpu_struct || !gpu_struct->has_gpu_data()) {
373 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
374 "Node context does not provide GpuStructuredData");
375 return;
376 }
377
378 auto data = gpu_struct->gpu_data();
379 ensure_buffer_capacity(binding, data.size_bytes());
380 upload_to_gpu(data.data(), data.size_bytes(), binding.gpu_buffer, nullptr);
381 break;
382 }
383 }

386void DescriptorBindingsProcessor::ensure_buffer_capacity(
387 DescriptorBinding& binding,
388 size_t required_size)
389{
390 if (required_size <= binding.gpu_buffer->get_size_bytes()) {
391 return;
392 }
393
394 size_t new_size = required_size * 3 / 2;
395
396 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
397 "Resizing descriptor buffer '{}': {} {} bytes",
398 binding.descriptor_name, binding.buffer_size, new_size);
399
400 binding.gpu_buffer->resize(new_size, false);
401
402 binding.buffer_size = new_size;
403
404 m_needs_descriptor_rebuild = true;
405}

407std::shared_ptr<VKBuffer> DescriptorBindingsProcessor::create_descriptor_buffer(
408 size_t size,
409 vk::DescriptorType type)
410{
411 VKBuffer::Usage usage {};
412
413 if (type == vk::DescriptorType::eUniformBuffer) {
414 usage = VKBuffer::Usage::UNIFORM;
415 } else { // SSBO
416 usage = VKBuffer::Usage::COMPUTE;
417 }
418
419 auto buffer = std::make_shared<VKBuffer>(
420 size,
421 usage,
422 Kakshya::DataModality::UNKNOWN);
423
424 auto buffer_service = Registry::BackendRegistry::instance()
425 .get_service<Registry::Service::BufferService>();
426
427 if (!buffer_service) {
428 error<std::runtime_error>(
429 Journal::Component::Buffers,
430 Journal::Context::BufferProcessing,
431 std::source_location::current(),
432 "create_descriptor_buffer requires a valid BufferService");
433 }
434
435 buffer_service->initialize_buffer(buffer);
436
437 return buffer;
438}
