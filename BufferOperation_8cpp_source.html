243bool BufferOperation::is_capture_phase_operation(const BufferOperation& op)
244{
245 if (op.get_execution_phase() == BufferOperation::ExecutionPhase::CAPTURE) {
246 return true;
247 }
248 if (op.get_execution_phase() == BufferOperation::ExecutionPhase::PROCESS) {
249 return false;
250 }
251
252 switch (op.get_type()) {
253 case BufferOperation::OpType::CAPTURE:
254 return true;
255 case BufferOperation::OpType::MODIFY:
256 return op.is_streaming();
257 default:
258 return false;
259 }
260}

262bool BufferOperation::is_process_phase_operation(const BufferOperation& op)
263{
264 if (op.get_execution_phase() == BufferOperation::ExecutionPhase::PROCESS) {
265 return true;
266 }
267 if (op.get_execution_phase() == BufferOperation::ExecutionPhase::CAPTURE) {
268 return false;
269 }
270
271 switch (op.get_type()) {
272 case BufferOperation::OpType::MODIFY:
273 return !op.is_streaming();
274 case BufferOperation::OpType::TRANSFORM:
275 case BufferOperation::OpType::ROUTE:
276 case BufferOperation::OpType::LOAD:
277 case BufferOperation::OpType::DISPATCH:
278 case BufferOperation::OpType::FUSE:
279 return true;
280 case BufferOperation::OpType::CAPTURE:
281 case BufferOperation::OpType::CONDITION:
282 case BufferOperation::OpType::BRANCH:
283 case BufferOperation::OpType::SYNC:
284 default:
285 return false;
286 }
