17/**
18 * @brief Result of make_collapsible().
19 *
20 * Owns all Forma constructs produced for one collapsible header. The header
21 * element is registered in the layer; any body elements the caller adds via
22 * layer.relate(header_id, body_id) will be shown and hidden automatically
23 * when the open state changes. cursor_out is positioned at the bottom of
24 * the header strip, ready for the first body element.
25 */
26struct Collapsible {
27 /// @brief Element id of the header strip. Pass to layer.relate() and ctx.on_press().
28 uint32_t header_id {};
29
30 /// @brief Open/closed state. Write to toggle; the sync loop picks up changes.
31 std::shared_ptr<MappedState<bool>> open;
32
33 /// @brief The FormaBuffer backing the header geometry.
34 std::shared_ptr<Buffers::FormaBuffer> buf;
35
36 /// @brief Cursor positioned at the bottom edge of the header, ready for body elements.
37 LayoutCursor cursor_out;
38
39 /**
40 * @brief Relate a body element to this collapsible and sync its visibility
41 * to the current open state.
42 *
43 * Replaces a manual layer.relate(header_id, body_id) call. Ensures the
44 * body's render state matches @ref open at attach time, so callers do not
45 * need to track the initial toggle state.
46 *
47 * @param layer Layer the body element lives on.
48 * @param body_id Element id returned from layer.add(...).
49 */
50 void attach(Layer& layer, uint32_t body_id) const
51 {
52 layer.relate(header_id, body_id);
53 layer.set_visible(body_id, open->value);
54 }
55};

57/**
58 * @brief Geometry function for a collapsible header strip.
59 *
60 * When @p label is null: emits 6 vertices (background quad only, weight=0).
61 * When @p label is non-null: emits 12 vertices — background quad (weight=0)
62 * followed by a full-cover overlay quad (weight=1, white, UVs [0,1]) that
63 * samples the label image via the multi-texture shader.
64 *
65 * Vertex layout matches VertexLayout::for_meshes() (stride 60):
66 * offset 0 position vec3
67 * offset 12 color vec3
68 * offset 24 weight float
69 * offset 28 uv vec2
70 *
71 * @param cursor_in Shared cursor state. top = cursor_in->value + row_h,
72 * bot = cursor_in->value (post-advance convention).
73 * @param x_min Left edge in NDC.
74 * @param x_max Right edge in NDC.
75 * @param row_h Strip height in NDC units.
76 * @param color_closed Background color when closed.
77 * @param color_open Background color when open.
78 * @param label Optional GPU image for the label overlay. nullptr = no overlay.
79 */
80[[nodiscard]] inline GeometryFn<bool> collapsible_header_geom(
81 float y_top,
82 float x_min,
83 float x_max,
84 float row_h,
85 glm::vec3 color_closed = glm::vec3(0.25F),
86 glm::vec3 color_open = glm::vec3(0.35F),
87 const std::shared_ptr<Core::VKImage>& label = nullptr)
88{
89 const bool has_label = label != nullptr;
90
91 return [y_top, x_min, x_max, row_h, color_closed, color_open, has_label](
92 bool open, std::vector<uint8_t>& out, Element& el) {
93 const float top = y_top;
94 const float bot = y_top - row_h;
95 const glm::vec3 col = open ? color_open : color_closed;
96
97 const size_t stride = Kakshya::VertexLayout::for_meshes().stride_bytes;
98 const size_t vertex_count = has_label ? 12 : 6;
99
100 out.assign(vertex_count * stride, 0);
101
102 const glm::vec3 bl { x_min, bot, 0.F };
103 const glm::vec3 br { x_max, bot, 0.F };
104 const glm::vec3 tl { x_min, top, 0.F };
105 const glm::vec3 tr { x_max, top, 0.F };
106 constexpr glm::vec3 white { 1.F, 1.F, 1.F };
107
108 auto write = [&](size_t idx, glm::vec3 pos, glm::vec3 c, float w, glm::vec2 uv) {
109 auto* v = out.data() + idx * stride;
110 std::memcpy(v, &pos, 12);
111 std::memcpy(v + 12, &c, 12);
112 std::memcpy(v + 24, &w, 4);
113 std::memcpy(v + 28, &uv, 8);
114 };
115
116 write(0, bl, col, 0.F, {});
117 write(1, br, col, 0.F, {});
118 write(2, tl, col, 0.F, {});
119 write(3, br, col, 0.F, {});
120 write(4, tr, col, 0.F, {});
121 write(5, tl, col, 0.F, {});
122
123 if (has_label) {
124 write(6, bl, white, 1.F, { 0.F, 1.F });
125 write(7, br, white, 1.F, { 1.F, 1.F });
126 write(8, tl, white, 1.F, { 0.F, 0.F });
127 write(9, br, white, 1.F, { 1.F, 1.F });
128 write(10, tr, white, 1.F, { 1.F, 0.F });
129 write(11, tl, white, 1.F, { 0.F, 0.F });
130 }
131
132 el.bounds_hint = Kinesis::AABB2D {
133 .min = { x_min, bot },
134 .max = { x_max, top },
135 };
136 };
137}

139/**
140 * @brief Construct a collapsible header strip.
141 *
142 * Creates and registers a FormaBuffer, constructs a Mapped<bool> whose
143 * geometry function produces the header quad, registers it with the Bridge
144 * sync loop, and wires the on_press toggle via @p ctx. Caller adds body
145 * elements to @p layer and calls layer.relate(col.header_id, body_id) for
146 * each one; set_visible cascades to all related ids on toggle.
147 *
148 * @param window Target window for rendering.
149 * @param layer Layer to register the header element on.
150 * @param ctx Context to wire the left-click toggle handler.
151 * @param bridge Bridge owning the sync loop.
152 * @param cursor Layout cursor. Advanced by @p row_h on return.
153 * @param x_min Left edge in NDC.
154 * @param x_max Right edge in NDC.
155 * @param row_h Header strip height in NDC units.
156 * @param initially_open Starting toggle state.
157 * @param color_closed Background color when closed.
158 * @param color_open Background color when open.
159 * @param label Optional GPU image overlaid on the header as a label.
160 * When provided, the header uses the multi-texture shader;
161 * when null, a plain color-only shader is used.
162 * @return Fully constructed Collapsible. cursor_out is below the header strip.
163 */
164[[nodiscard]] inline Collapsible make_collapsible(
165 const std::shared_ptr<Core::Window>& window,
166 Layer& layer,
167 Context& ctx,
168 Bridge& bridge,
169 LayoutCursor& cursor,
170 float x_min,
171 float x_max,
172 float row_h,
173 bool initially_open = true,
174 glm::vec3 color_closed = glm::vec3(0.25F),
175 glm::vec3 color_open = glm::vec3(0.35F),
176 std::shared_ptr<Core::VKImage> label = nullptr)
177{
178 const size_t vertex_count = label ? 12 : 6;
179 const size_t cap = vertex_count * Kakshya::VertexLayout::for_meshes().stride_bytes;
180
181 auto buf = label
182 ? create_buffer(window, cap,
183 Graphics::PrimitiveTopology::TRIANGLE_LIST,
184 std::vector<std::pair<std::string, std::shared_ptr<Core::VKImage>>> {
185 { "text", label } })
186 : create_buffer(window, cap,
187 Graphics::PrimitiveTopology::TRIANGLE_LIST);
188
189 const float y_top = cursor.y();
190 cursor.advance(row_h);
191
192 auto open_state = std::make_shared<MappedState<bool>>();
193 open_state->write(initially_open);
194
195 Mapped<bool> mapped;
196 mapped.state = open_state;
197 mapped.geometry_fn = collapsible_header_geom(
198 y_top, x_min, x_max, row_h, color_closed, color_open, label);
199
200 mapped.element.buffer = buf;
201 mapped.element.bounds_hint = Kinesis::AABB2D {
202 .min = { x_min, cursor.y() },
203 .max = { x_max, cursor.y() + row_h },
204 };
205
206 const uint32_t hid = layer.add(mapped.element);
207 mapped.element.id = hid;
208 open_state->id = hid;
209 bridge.register_element(std::move(mapped));
210
211 ctx.on_press(hid, IO::MouseButtons::Left,
212 [open = open_state, &layer, hid](uint32_t, glm::vec2) {
213 const bool next = !open->value;
214 open->write(next);
215 for (auto rel_id : layer.related_ids(hid))
216 layer.set_visible(rel_id, next);
217 });
218
219 return Collapsible {
220 .header_id = hid,
221 .open = open_state,
222 .buf = buf,
223 .cursor_out = LayoutCursor(cursor.y()),
224 };
225}
