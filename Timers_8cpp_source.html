148void TemporalActivation::cleanup_current_operation()
149{
150 switch (m_active_type) {
151 case ActiveType::NODE:
152 if (m_current_node) {
153 if (m_channels.empty()) {
154 m_node_graph_manager.get_root_node(m_node_token, 0).unregister_node(m_current_node);
155 }
156 for (auto channel : m_channels) {
157 if (m_current_node->is_used_by_channel(channel)) {
158 m_node_graph_manager.get_root_node(m_node_token, channel).unregister_node(m_current_node);
159 }
160 }
161 m_current_node = nullptr;
162 m_channels.clear();
163 }
164 break;
165
166 case ActiveType::NETWORK:
167 if (m_current_network) {
168 for (const auto& ch : m_channels) {
169 m_current_network->remove_channel_usage(ch);
170 }
171
172 m_node_graph_manager.remove_network(m_current_network, m_node_token);
173 m_current_network = nullptr;
174 }
175 break;
176
177 case ActiveType::BUFFER:
178 if (m_current_buffer) {
179 for (auto channel : m_channels) {
180 m_buffer_manager.remove_buffer(m_current_buffer, m_buffer_token, channel);
181 }
182 m_current_buffer = nullptr;
183 }
184 break;
185
186 case ActiveType::NONE:
187 break;
188 }
189
190 m_active_type = ActiveType::NONE;
191}
