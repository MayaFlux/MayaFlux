100void Engine::Init(const GlobalStreamInfo& streamInfo, const GlobalGraphicsConfig& graphics_config, const GlobalInputConfig& input_config)
101{
102 MF_PRINT(Journal::Component::Core, Journal::Context::Init, "Engine initializing");
103 m_stream_info = streamInfo;
104 m_graphics_config = graphics_config;
105 m_input_config = input_config;
106
107 m_scheduler = std::make_shared<Vruta::TaskScheduler>(m_stream_info.sample_rate);
108 m_event_manager = std::make_shared<Vruta::EventManager>();
109 m_input_manager = std::make_shared<InputManager>();
110
111 m_buffer_manager = std::make_shared<Buffers::BufferManager>(
112 m_stream_info.output.channels,
113 m_stream_info.input.enabled ? m_stream_info.input.channels : 0,
114 m_stream_info.sample_rate, m_stream_info.buffer_size);
115
116 m_node_graph_manager = std::make_shared<Nodes::NodeGraphManager>(m_stream_info.sample_rate, m_stream_info.buffer_size);
117
118 if (m_graphics_config.windowing_backend != GlobalGraphicsConfig::WindowingBackend::NONE) {
119 m_window_manager = std::make_shared<WindowManager>(m_graphics_config);
120 } else {
121 MF_WARN(Journal::Component::Core, Journal::Context::Init, "No windowing backend selected - running in audio-only mode");
122 }
123
124 m_subsystem_manager = std::make_shared<SubsystemManager>(
125 m_node_graph_manager, m_buffer_manager, m_scheduler, m_window_manager, m_input_manager);
126
127 m_subsystem_manager->create_audio_subsystem(m_stream_info);
128
129 m_subsystem_manager->create_graphics_subsystem(m_graphics_config);
130
131 m_subsystem_manager->create_input_subsystem(m_input_config);
132
133 m_buffer_manager->initialize_buffer_service();
134
135 m_is_initialized = true;
136
137 MF_PRINT(Journal::Component::Core, Journal::Context::Init, "Audio backend: RtAudio, Sample rate: {}", m_stream_info.sample_rate);
138}

168bool Engine::is_running() const
169{
170 if (!m_is_initialized || m_is_paused) {
171 return false;
172 }
173
174 if (m_is_initialized) {
175 auto status = m_subsystem_manager->query_subsystem_status();
176 for (const auto& [type, readiness] : status) {
177 const auto& [is_ready, is_running] = readiness;
178 if (is_ready && is_running) {
179 return true;
180 }
181 }
182 }
183 return false;

220void Engine::run_macos_event_loop()
221{
222 CFRunLoopRef runLoop = CFRunLoopGetMain();
223
224 dispatch_source_t stdinSource = dispatch_source_create(
225 DISPATCH_SOURCE_TYPE_READ,
226 STDIN_FILENO,
227 0,
228 dispatch_get_main_queue());
229
230 dispatch_source_set_event_handler(stdinSource, ^{

234 request_shutdown();

239
240 double timeout_seconds = 1.0 / static_cast<double>(m_graphics_config.target_frame_rate);

244 m_graphics_config.target_frame_rate);
245
246 while (!is_shutdown_requested()) {
247 CFRunLoopRunInMode(kCFRunLoopDefaultMode, timeout_seconds, false);
248 }
249
250 dispatch_source_cancel(stdinSource);

259void Engine::run_windows_event_loop()
260{
261 MSG msg;
262 PeekMessage(&msg, NULL, WM_USER, WM_USER, PM_NOREMOVE);
263
264 auto stdinSource = std::thread([this]() {
265 std::cin.get();
266 request_shutdown();
267 });
268
269 MF_PRINT(Journal::Component::Core, Journal::Context::Runtime,
270 "Main thread event loop running");
271
272 while (!is_shutdown_requested()) {
273 while (PeekMessage(&msg, NULL, 0, 0, PM_REMOVE)) {
274 if (msg.message == WM_QUIT) {
275 request_shutdown();
276 break;
277 }
278
279 if (msg.message == MAYAFLUX_WM_DISPATCH) {
280 auto* task = reinterpret_cast<std::function<void()>*>(msg.lParam);
281 if (task) {
282 (*task)();
283 delete task;
284 }
285 } else {
286 TranslateMessage(&msg);
287 DispatchMessage(&msg);
288 }
289 }
290
291 std::this_thread::sleep_for(std::chrono::milliseconds(1));
292 }
293
294 if (stdinSource.joinable()) {
295 stdinSource.detach();
296 }
297
298 MF_INFO(Journal::Component::Core, Journal::Context::Runtime,
299 "Main thread event loop exiting");
300}

303void Engine::End()
304{
305 if (!m_is_initialized)
306 return;
307
308 if (m_node_graph_manager) {
309 m_node_graph_manager->terminate_active_processing();
310 }
311
312 if (m_subsystem_manager) {
313 m_subsystem_manager->stop();
314 }
315
316 if (m_scheduler) {
317 m_scheduler->terminate_all_tasks();
318 }
319
320 if (m_buffer_manager) {
321 m_buffer_manager->terminate_active_buffers();
322 m_buffer_manager.reset();
323 }
324
325 if (m_window_manager) {
326 m_window_manager->set_terminate();
327 auto windows = m_window_manager->get_windows();
328 for (auto& window : windows) {
329 m_window_manager->destroy_window(window, true);
330 }
331 m_window_manager.reset();
332 }
333
334 if (m_node_graph_manager) {
335 m_node_graph_manager.reset();
336 }
337
338 if (m_subsystem_manager) {
339 m_subsystem_manager->shutdown();
340 }
341
342 m_is_initialized = false;
343 m_is_paused = false;
344}
