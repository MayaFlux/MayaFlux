3/**
4 * @file VisionOp.hpp
5 * @brief Declarative description of a Kinesis::Vision processing sequence.
6 *
7 * VisionStep names one operation and carries the parameters needed to invoke
8 * it. VisionSequence is an ordered list of steps. Both are pure value types
9 * with no MayaFlux dependencies.
10 *
11 * Processors (VisionDataProcessor, VisionBufferProcessor) accept a
12 * VisionSequence at construction and execute it each cycle without knowing
13 * which specific algorithms are involved. No per-algorithm processor subclass
14 * is needed.
15 */
16

140/**
141 * @brief One step in a VisionSequence: an op and its parameters.
142 */
143struct VisionStep {
144 VisionOp op;
145 VisionParams params { std::monostate {} };

148/**
149 * @brief Ordered sequence of VisionSteps describing a complete vision pipeline.

151 * Constructed via the fluent VisionSequence::Builder.

153struct VisionSequence {

306 [[nodiscard]] VisionSequence build()
307 {
308 VisionSequence seq { .steps = std::move(m_steps) };
309 for (size_t i = 0; i < seq.steps.size(); ++i) {
310 if (seq.steps[i].op == VisionOp::TrackKeypoints)
311 seq.tracks_keypoints = true;
312 if (i + 1 < seq.steps.size()
313 && seq.steps[i].op == VisionOp::ExtractPeaks
314 && seq.steps[i + 1].op == VisionOp::TrackKeypoints)
315 seq.track_follows_peaks = true;
316 }
317 return seq;
318 }

327 }
