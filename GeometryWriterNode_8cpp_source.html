46void GeometryWriterNode::set_vertex_data(const void* data, size_t size_bytes)
47{
48 if (!data || size_bytes == 0) {
49 MF_ERROR(Journal::Component::Nodes, Journal::Context::NodeProcessing,
50 "Cannot set vertex data: null data or zero size");
51 return;
52 }
53
54 if (m_vertex_stride == 0) {
55 MF_ERROR(Journal::Component::Nodes, Journal::Context::NodeProcessing,
56 "Cannot set vertex data: stride is 0");
57 return;
58 }
59
60 auto expected_vertex_count = static_cast<uint32_t>(size_bytes / m_vertex_stride);
61
62 if (size_bytes % m_vertex_stride != 0) {
63 MF_WARN(Journal::Component::Nodes, Journal::Context::NodeProcessing,
64 "Vertex data size {} is not multiple of stride {}",
65 size_bytes, m_vertex_stride);
66 }
67
68 size_t required_floats = size_bytes / sizeof(float);
69 if (m_vertex_buffer.size() < required_floats) {
70 m_vertex_buffer.resize(required_floats);
71 }
72
73 std::memcpy(m_vertex_buffer.data(), data, size_bytes);
74 m_vertex_count = expected_vertex_count;
75 m_needs_layout_update = true;
76 m_vertex_data_dirty = true;

136{

228std::vector<LineVertex> GeometryWriterNode::expand_lines_to_triangles(
229 const std::vector<LineVertex>& line_segment)
230{
231 if (line_segment.size() < 2) {
232 return {};
233 }
234
235 std::vector<LineVertex> triangles;
236 triangles.reserve((line_segment.size() - 1) * 6);
237
238 for (size_t i = 0; i + 1 < line_segment.size(); ++i) {
239 const auto& v0 = line_segment[i];
240 const auto& v1 = line_segment[i + 1];
241
242 glm::vec2 p0_2d = glm::vec2(v0.position.x, v0.position.y);
243 glm::vec2 p1_2d = glm::vec2(v1.position.x, v1.position.y);
244
245 glm::vec2 dir = p1_2d - p0_2d;
246 float len = glm::length(dir);
247
248 if (len < 1e-6F) {
249 continue;
250 }
251
252 dir = dir / len;
253 glm::vec2 normal = glm::vec2(-dir.y, dir.x);
254
255 float t0 = v0.thickness * 0.005F;
256 float t1 = v1.thickness * 0.005F;
257
258 glm::vec3 p0_top = glm::vec3(p0_2d + normal * t0, v0.position.z);
259 glm::vec3 p0_bot = glm::vec3(p0_2d - normal * t0, v0.position.z);
260 glm::vec3 p1_top = glm::vec3(p1_2d + normal * t1, v1.position.z);
261 glm::vec3 p1_bot = glm::vec3(p1_2d - normal * t1, v1.position.z);
262
263 triangles.push_back({ p0_top, v0.color, v0.thickness,
264 glm::vec2(0.0F, 0.0F) });
265 triangles.push_back({ p0_bot, v0.color, v0.thickness,
266 glm::vec2(0.0F, 1.0F) });
267 triangles.push_back({ p1_top, v1.color, v1.thickness,
268 glm::vec2(1.0F, 0.0F) });
269
270 triangles.push_back({ p0_bot, v0.color, v0.thickness,
271 glm::vec2(0.0F, 1.0F) });
272 triangles.push_back({ p1_bot, v1.color, v1.thickness,
273 glm::vec2(1.0F, 1.0F) });
274 triangles.push_back({ p1_top, v1.color, v1.thickness,
275 glm::vec2(1.0F, 0.0F) });
276 }
277
278 return triangles;
279}
