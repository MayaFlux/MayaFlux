28/**
29 * @class Bridge
30 * @brief Two-way binding orchestrator for Forma elements.
31 *
32 * One Bridge instance serves the full application. It owns all inbound
33 * GraphicsRoutine tasks and all outbound FormaBindingsProcessor instances.
34 * Layer and Context are created via the Forma free functions and passed in
35 * — Bridge does not own or create them.
36 *
37 * Inbound (what drives an element's MappedState each frame):
38 * - bind(id, Node) — reads Node::get_last_output() via GraphicsRoutine
39 * - bind(id, lambda) — calls std::function<float()> via GraphicsRoutine
40 *
41 * Outbound (where the element's value goes each frame):
42 * - write(id, ShaderProcessor, offset) — push constant staging
43 * - write(id, descriptor_name, ...) — descriptor binding
44 * - write(id, AudioWriteProcessor) — audio buffer
45 * - write(id, GeometryWriteProcessor) — vertex buffer
46 * - write(id, Constant) — node graph value carrier
47 *
48 * All overloads also accept shared_ptr<MappedState<T>> in place of the
49 * element id. These resolve to the state->id.
50 * duplicated logic.
51 *
52 * Usage:
53 * @code
54 * auto [layer, ctx] = Forma::create_layer(window, "hud", event_manager);
55 * auto el = Forma::create_element<float>(*layer, window, bm, geom, 0.5f);
56 *
57 * Bridge bridge(scheduler, buffer_manager);
58 * bridge.bind(el.state, envelope);
59 * bridge.write(el.state, compute_proc, offsetof(PC, cutoff));
60 *
61 * ctx->on_press(el.element.id, IO::MouseButtons::LEFT,
62 * [](uint32_t, glm::vec2){});
63 * @endcode
64 */

243 /**
244 * @brief Register a MappedState<T> so that MappedState overloads and
245 * outbound bindings can resolve to the correct element id and reader.
246 *
247 * Called by Forma::create_element. The reader closes over the MappedState
248 * and is stored type-erased as std::function<float()>.
249 *
250 * @tparam T MappedState value type.
251 * @param state MappedState to register.
252 * @param id Element id from Layer::add.
253 * @param buffer FormaBuffer the element renders into.
254 * @param project Optional T -> float projection. Identity cast if empty.
255 */
256 template <typename T>
257 void register_element(
258 std::shared_ptr<MappedState<T>> state,
259 uint32_t id,
260 std::shared_ptr<Buffers::FormaBuffer> buffer,
261 std::function<float(T)> project = {})
262 {
263 std::function<float()> reader = project
264 ? std::function<float()>([s = state, p = std::move(project)] {
265 return p(s->value);
266 })
267 : std::function<float()>([s = state] {
268 return static_cast<float>(s->value);
269 });
270
271 std::function<void(float)> writer = [s = state](float v) {
272 s->write(static_cast<T>(v));
273 };
274
275 state->id = id;
276 m_records[id] = ElementRecord {
277 .buffer = std::move(buffer),
278 .bindings = nullptr,
279 .reader = std::move(reader),
280 .writer = std::move(writer),
281 .inbound_task = {},
282 .outbound_tasks = {},
283 };
284 }

286 /**
287 * @brief Register a fully constructed Mapped<T> and own its sync loop.
288 *
289 * Delegates record and reader/writer setup to the existing
290 * register_element overload, then spawns a per-frame GraphicsRoutine
291 * that calls mapped.sync(). The routine is stored in outbound_tasks
292 * and cancelled by unbind().
293 *
294 * @p mapped must outlive the Bridge registration.
295 *
296 * @tparam T MappedState value type.
297 * @param mapped Fully constructed Mapped<T>, typically from create_element.
298 * @param project Optional T -> float projection for outbound readers.
299 * Identity cast used if empty.
300 */
301 template <typename T>
302 void register_element(Mapped<T> mapped, std::function<float(T)> project = {})
303 {
304 register_element(mapped.state, mapped.element.id, mapped.element.buffer, std::move(project));
305 spawn_sync(mapped.element.id, [m = std::move(mapped)]() mutable { m.sync(); });
306 }
