94bool AudioBuffer::read_once(std::shared_ptr<AudioBuffer> buffer, bool force)
95{
96 if (buffer && buffer->get_num_samples() == m_num_samples) {
97 if (m_is_processing.load() || buffer->is_processing()) {
98 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
99 "read_once: Attempting to read from an audio buffer while it is being processed.");
100
101 if (!force) {
102 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
103 "read_once: Skipping read due to ongoing processing.");
104 return false;
105 }
106 MF_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
107 "read_once: Forcing read despite ongoing processing. This may lead to data corruption.");
108 }
109 m_data = buffer->get_data();
110 m_has_data = true;
111 return true;
112 }
113
114 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
115 "read_once: Buffer read failed due to size mismatch or null buffer.");
116
117 return false;
118}
