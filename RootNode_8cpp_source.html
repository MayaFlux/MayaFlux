118double RootNode::process_sample()
119{
120 if (!preprocess())
121 return 0.;
122
123 auto sample = 0.;
124
125 for (auto& node : m_Nodes) {
126 if (!node)
127 continue;
128
129 uint32_t state = node->m_state.load();
130 double node_output = 0.0;
131
132 if (!(state & NodeState::PROCESSED)) {
133 auto generator = std::dynamic_pointer_cast<Nodes::Generator::Generator>(node);
134 if (generator && generator->should_mock_process()) {
135 generator->process_sample();
136 } else {
137 node_output = node->process_sample();

140 } else {
141 node_output = node->get_last_output();

145 node_output *= node->get_routing_state().amount[m_channel];
146 }

149 }

206void RootNode::process_pending_operations()
207{
208
209 for (auto& m_pending_op : m_pending_ops) {
210 if (m_pending_op.active.load(std::memory_order_acquire)) {
211 auto& op = m_pending_op;
212 uint32_t state = op.node->m_state.load();
213
214 if (!(state & NodeState::ACTIVE)) {
215 m_Nodes.push_back(op.node);
216
217 state &= ~static_cast<uint32_t>(NodeState::INACTIVE);
218 state |= static_cast<uint32_t>(NodeState::ACTIVE);
219 atomic_set_flag_strong(op.node->m_state, static_cast<NodeState>(state));
220 } else if (state & NodeState::PENDING_REMOVAL) {
221 auto it = m_Nodes.begin();
222 while (it != m_Nodes.end()) {
223 if ((*it).get() == op.node.get()) {
224 it = m_Nodes.erase(it);
225 break;
226 }
227 ++it;
228 }
229 op.node->reset_processed_state();
230
231 state &= ~static_cast<uint32_t>(NodeState::PENDING_REMOVAL);
232 state &= ~static_cast<uint32_t>(NodeState::ACTIVE);
233 state |= static_cast<uint32_t>(NodeState::INACTIVE);
234 atomic_set_flag_strong(op.node->m_state, static_cast<NodeState>(state));
235 }
236
237 op.node.reset();
238 op.active.store(false, std::memory_order_release);
239 m_pending_count.fetch_sub(1, std::memory_order_relaxed);
240 }
