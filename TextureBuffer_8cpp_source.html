71void TextureBuffer::setup_rendering(const RenderConfig& user_config)
72{
73 if (!user_config.vertex_shader.empty()) {
74 m_render_config.vertex_shader = user_config.vertex_shader;
75 }
76 if (!user_config.fragment_shader.empty()) {
77 m_render_config.fragment_shader = user_config.fragment_shader;
78 }
79 if (!user_config.default_texture_binding.empty()) {
80 m_render_config.default_texture_binding = user_config.default_texture_binding;
81 }
82 if (user_config.topology != Portal::Graphics::PrimitiveTopology::TRIANGLE_STRIP) {
83 MF_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
84 "TextureBuffer only supports TRIANGLE_STRIP topology. Ignoring provided topology.");
85 } else {
86 if (user_config.topology != m_render_config.topology) {
87 m_render_config.topology = user_config.topology;
88 }
89 }
90 m_render_config.target_window = user_config.target_window;
91
92 if (!user_config.additional_textures.empty()) {
93 for (const auto& [name, texture] : user_config.additional_textures) {
94 m_render_config.additional_textures.emplace_back(name, texture);
95 }
96 }
97
98 if (!m_render_processor) {
99 ShaderConfig shader_config { m_render_config.vertex_shader };
100 shader_config.bindings[m_render_config.default_texture_binding] = ShaderBinding(
101 0, 0, vk::DescriptorType::eCombinedImageSampler);
102
103 uint32_t binding_index = 1;
104 for (const auto& [name, _] : m_render_config.additional_textures) {
105 shader_config.bindings[name] = ShaderBinding(
106 0, binding_index++, vk::DescriptorType::eCombinedImageSampler);
107 }
108
109 m_render_processor = std::make_shared<RenderProcessor>(shader_config);
110 }
111
112 m_render_processor->set_fragment_shader(m_render_config.fragment_shader);
113 m_render_processor->set_target_window(m_render_config.target_window, std::dynamic_pointer_cast<VKBuffer>(shared_from_this()));
114 m_render_processor->set_primitive_topology(m_render_config.topology);
115
116 m_render_processor->bind_texture(m_render_config.default_texture_binding, get_texture());
117
118 for (const auto& [name, texture] : m_render_config.additional_textures) {
119 m_render_processor->bind_texture(name, texture);
120 }
121
122 get_processing_chain()->add_final_processor(m_render_processor, shared_from_this());
123}

252void TextureBuffer::generate_quad_with_transform()
253{
254 if (m_uses_custom_vertices) {
255 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
256 "TextureBuffer: using custom vertices, skipping transform");
257 return;
258 }
259
260 float cos_rot = std::cos(m_rotation);
261 float sin_rot = std::sin(m_rotation);
262
263 std::vector<QuadVertex> transformed(4);
264 for (size_t i = 0; i < 4; ++i) {
265 glm::vec3 pos = base_quad[i].position;
266
267 pos.x *= m_scale.x;
268 pos.y *= m_scale.y;
269
270 float rotated_x = pos.x * cos_rot - pos.y * sin_rot;
271 float rotated_y = pos.x * sin_rot + pos.y * cos_rot;
272
273 pos.x = rotated_x + m_position.x;
274 pos.y = rotated_y + m_position.y;
275
276 transformed[i].position = pos;
277 transformed[i].texcoord = base_quad[i].texcoord;
278 }
279
280 m_vertex_bytes.resize(transformed.size() * sizeof(QuadVertex));
281 std::memcpy(m_vertex_bytes.data(), transformed.data(), m_vertex_bytes.size());
282
283 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
284 "TextureBuffer: regenerated quad with transform (pos={},{}, scale={},{}, rot={})",
285 m_position.x, m_position.y, m_scale.x, m_scale.y, m_rotation);
286}
