29/**
30 * @enum GranularOutput
31 * @brief Selects the terminal output type produced by make_granular_matrix.
32 *
33 * REGION_GROUP — pipeline terminates with a sorted RegionGroup (default).
34 * Use when you need to inspect, analyse, or further transform
35 * the grain pool before committing to audio.
36 *
37 * CONTAINER — appends a reconstruct rule that stitches sorted grains into
38 * a SoundFileContainer. Use for the offline playback workflow
39 * where container-in container-out is all that is needed.
40 *
41 * CONTAINER_ADDITIVE — overlap-add reconstruct: grains accumulated at hop_size
42 * intervals with optional per-grain tapering.
43 */
44enum class GranularOutput : uint8_t {
45 REGION_GROUP,
46 CONTAINER,
47 CONTAINER_ADDITIVE

121/**
122 * @class AttributeOp
123 * @brief Computes a scalar attribute per grain and writes it onto each Region.
124 *
125 * Attribution is resolved in the following priority order:
126 *
127 * 1. Span lambda : set via set_parameter("attribute_executor", AttributeExecutor)
128 * or present in ExecutionContext. For fully custom math.
129 *
130 * 2. Direct analyzer : set via set_parameter("analyzer", shared_ptr<UniversalAnalyzer<...>>)
131 * paired with set_parameter("analyzer_qualifier", std::string).
132 * Any existing concrete analyzer can be supplied pre-configured.
133 *
134 * 3. Analysis type : set via set_parameter("analysis_type", AnalysisType) with
135 * optional set_parameter("analyzer_qualifier", std::string). AttributeOp
136 * constructs a default-configured analyzer internally.
137 *
138 * Parameters common to all paths:
139 * - feature_key std::string attribute name written on each grain (default "feature")
140 * - channel uint32_t source channel index (default 0)
141 *
142 * Qualifier strings map directly to scalar fields of the analysis result struct.
143 * The alias "rms" is accepted for FEATURE and resolves to "mean_energy".
144 * The alias "mean" is accepted for STATISTICAL and resolves to "mean_stat".
145 *
146 * FEATURE qualifiers (EnergyAnalysis):
147 * - "mean_energy" (default, alias: "rms")
148 * - "max_energy"
149 * - "min_energy"
150 * - "variance"
151 * - "dynamic_range"
152 * - "event_count"
153 * - "window_count"
154 *
155 * STATISTICAL qualifiers (StatisticalAnalysis):
156 * - "mean_stat" (default, alias: "mean")
157 * - "max_stat"
158 * - "min_stat"
159 * - "variance"
160 * - "std_dev"
161 * - "skewness"
162 * - "kurtosis"
163 * - "median"
164 * - "window_count"
165 */

265/**
266 * @brief Construct an ExecutionContext for the granular pipeline using a built-in AnalysisType.
267 *
268 * AttributeOp will internally construct a default-configured analyzer for the
269 * given type and extract the scalar named by qualifier. Omit qualifier to use
270 * the canonical default for that type.
271 *
272 * @param grain_size Samples per grain.
273 * @param hop_size Hop size between grain onsets.
274 * @param feature_key Attribute name written onto each grain Region.
275 * @param analysis_type Which analysis category to use for attribution. Supported: FEATURE, STATISTICAL.
276 * @param qualifier Scalar to extract from the analysis result. Empty uses type default.
277 * @param channel Source channel index.
278 * @param ascending Sort direction for the subsequent SortOp.
279 * @param gpu_sort_threshold Grain count at which SortOp switches to GPU path. 0 = CPU only.
280 * @return Populated ExecutionContext.
281 */
282[[nodiscard]] inline ExecutionContext make_granular_context(
283 uint32_t grain_size,
284 uint32_t hop_size,
285 const std::string& feature_key,
286 AnalysisType analysis_type,
287 const std::string& qualifier = {},
288 uint32_t channel = 0,
289 bool ascending = true,
290 uint32_t gpu_sort_threshold = 0)
291{
292 ExecutionContext ctx;
293 ctx.execution_metadata["grain_size"] = grain_size;
294 ctx.execution_metadata["hop_size"] = hop_size;
295 ctx.execution_metadata["channel"] = channel;
296 ctx.execution_metadata["feature_key"] = feature_key;
297 ctx.execution_metadata["ascending"] = ascending;
298 ctx.execution_metadata["gpu_sort_threshold"] = gpu_sort_threshold;
299 ctx.execution_metadata["analysis_type"] = analysis_type;
300 ctx.execution_metadata["analyzer_qualifier"] = qualifier;
301 return ctx;
302}

304/**

328 const std::string& qualifier = {},
329 uint32_t channel = 0,

335 ctx.execution_metadata["hop_size"] = hop_size;

338 ctx.execution_metadata["ascending"] = ascending;
339 ctx.execution_metadata["gpu_sort_threshold"] = gpu_sort_threshold;
