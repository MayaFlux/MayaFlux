105void NodeNetwork::request_reset_from_channel(uint32_t channel_id)
106{
107 if (channel_id >= 32)
108 return;
109
110 auto channel_bit = static_cast<uint32_t>(1ULL << channel_id);
111 uint32_t old_pending = m_pending_reset_mask.fetch_or(channel_bit,
112 std::memory_order_acq_rel);
113 uint32_t new_pending = old_pending | channel_bit;
114 uint32_t active_channels = m_channel_mask.load(std::memory_order_acquire);
115
116 if ((new_pending & active_channels) == active_channels && active_channels != 0) {
117 uint32_t expected = new_pending;
118 if (m_pending_reset_mask.compare_exchange_strong(expected, 0,
119 std::memory_order_acq_rel)) {
120 mark_processed(false);
121 }
122 }
123}
