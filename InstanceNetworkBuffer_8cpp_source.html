144size_t InstanceNetworkBuffer::estimate_vertex_bytes(
145 const std::shared_ptr<Nodes::Network::InstanceNetwork>& network,
146 float factor)
147{
148 constexpr size_t k_min = 64L * 1024;
149 if (!network || network->slot_count() == 0)
150 return k_min;
151
152 const auto& slots = network->slots();
153 if (!slots[0].node)
154 return k_min;
155
156 const size_t capacity = slots[0].node->get_vertex_buffer_size_bytes();
157 if (capacity == 0)
158 return k_min;
159
160 return static_cast<size_t>(static_cast<float>(capacity) * factor);
161}
