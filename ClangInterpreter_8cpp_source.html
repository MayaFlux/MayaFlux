42bool ClangInterpreter::initialize()
43{
44 LILA_INFO(Emitter::INTERPRETER, "Initializing Clang interpreter");
45
46#ifdef MAYAFLUX_PLATFORM_WINDOWS
47 llvm::sys::DynamicLibrary::LoadLibraryPermanently("msvcp140.dll");
48 llvm::sys::DynamicLibrary::LoadLibraryPermanently("vcruntime140.dll");
49 llvm::sys::DynamicLibrary::LoadLibraryPermanently("ucrtbase.dll");
50 llvm::sys::DynamicLibrary::LoadLibraryPermanently("MayaFluxLib.dll");
51#endif
52
53 llvm::InitializeNativeTarget();
54 llvm::InitializeNativeTargetAsmPrinter();
55 llvm::InitializeNativeTargetAsmParser();
56
57 m_impl->compile_flags.clear();
58 m_impl->compile_flags.emplace_back("-std=c++23");
59 m_impl->compile_flags.emplace_back("-DMAYASIMPLE");
60
61#ifdef MAYAFLUX_PLATFORM_LINUX
62 m_impl->compile_flags.emplace_back("-mcmodel=large");
63 m_impl->compile_flags.emplace_back("-fPIC");
64 m_impl->compile_flags.emplace_back("-fPIE");
65#endif
66
67 std::string pch_dir;
68 if (std::filesystem::exists(MayaFlux::Config::PCH_RUNTIME_PATH)) {
69 pch_dir = MayaFlux::Config::RUNTIME_DATA_DIR;
70 LILA_DEBUG(Emitter::INTERPRETER,
71 std::string("Using installed PCH from: ") + std::string(MayaFlux::Config::PCH_RUNTIME_PATH));
72
73 } else if (std::filesystem::exists(MayaFlux::Config::PCH_SOURCE_PATH)) {
74 pch_dir = std::string(MayaFlux::Config::SOURCE_DIR) + "/cmake";
75 LILA_DEBUG(Emitter::INTERPRETER,
76 std::string("Using source PCH from: ") + std::string(MayaFlux::Config::PCH_SOURCE_PATH));
77
78 } else {
79 m_last_error = "Cannot find pch.h in runtime or source locations";
80 LILA_ERROR(Emitter::INTERPRETER, m_last_error);
81 return false;
82 }
83
84 m_impl->compile_flags.push_back("-I" + pch_dir);
85
86 std::string resource_dir = MayaFlux::Platform::SystemConfig::get_clang_resource_dir();
87 if (!resource_dir.empty()) {
88 m_impl->compile_flags.push_back("-resource-dir=" + resource_dir);
89 LILA_DEBUG(Emitter::INTERPRETER,
90 std::string("Using clang resource dir: ") + resource_dir);
91 } else {
92 m_impl->compile_flags.emplace_back("-resource-dir=/usr/lib/clang/21");
93 LILA_WARN(Emitter::INTERPRETER,
94 "Using default clang resource dir: /usr/lib/clang/21");
95 }
96
97 auto system_includes = MayaFlux::Platform::SystemConfig::get_system_includes();
98 for (const auto& include : system_includes) {
99 m_impl->compile_flags.push_back("-isystem" + include);
100 }
101
102#ifdef MAYAFLUX_PLATFORM_MACOS
103 // CRITICAL: JIT uses Homebrew LLVM but needs macOS SDK for system headers
104 // Homebrew LLVM's libc++ requires pthread.h, sched.h, time.h, etc. from SDK
105 std::string sdk_path = MayaFlux::Platform::SystemConfig::get_macos_sdk_path();
106 if (!sdk_path.empty()) {
107 m_impl->compile_flags.push_back("-isysroot" + sdk_path);
108 LILA_DEBUG(Emitter::INTERPRETER, "Using macOS SDK: " + sdk_path);
109 } else {
110 LILA_WARN(Emitter::INTERPRETER,
111 "Could not find macOS SDK - JIT may fail to find system headers");
112 }
113#endif
114
115 for (const auto& path : m_impl->include_paths) {
116 m_impl->compile_flags.push_back("-I" + path);
117 }
118
119#ifdef MAYAFLUX_PLATFORM_WINDOWS
120 m_impl->compile_flags.emplace_back("-fno-function-sections");
121 m_impl->compile_flags.emplace_back("-fno-data-sections");
122 m_impl->compile_flags.emplace_back("-fno-unique-section-names");
123#endif
124
125 std::vector<const char*> args;
126 for (const auto& flag : m_impl->compile_flags) {
127 args.push_back(flag.c_str());
128 }
129
130 clang::IncrementalCompilerBuilder ICB;
131 ICB.SetCompilerArgs(args);
132
133 auto CI = ICB.CreateCpp();
134 if (!CI) {
135 m_last_error = "Failed to create CompilerInstance: " + llvm::toString(CI.takeError());
136 LILA_ERROR(Emitter::INTERPRETER, m_last_error);
137 return false;
138 }
139
140 auto interp = clang::Interpreter::create(std::move(*CI));
141 if (!interp) {
142 m_last_error = "Failed to create interpreter: " + llvm::toString(interp.takeError());
143 LILA_ERROR(Emitter::INTERPRETER, m_last_error);
144 return false;
145 }
146
147 m_impl->interpreter = std::move(*interp);
148
149 LILA_INFO(Emitter::INTERPRETER, "Clang interpreter created successfully");
150
151 auto result = m_impl->interpreter->ParseAndExecute(
152 "#include \"pch.h\"\n"
153 "#include \"Lila/LiveAid.hpp\"\n");
154
155 if (result) {
156 std::string warning = "Failed to load MayaFlux headers: " + llvm::toString(std::move(result));
157 LILA_WARN(Emitter::INTERPRETER, warning);
158 } else {
159 LILA_INFO(Emitter::INTERPRETER, "MayaFlux headers loaded successfully");
160 }
161
162 result = m_impl->interpreter->ParseAndExecute("std::cout << \"Ready for Live\" << std::flush;");
163
164 return true;
165}

167ClangInterpreter::EvalResult ClangInterpreter::eval(const std::string& code)
168{
169 EvalResult result;
170
171 if (!m_impl->interpreter) {
172 result.error = "Interpreter not initialized";
173 LILA_ERROR(Emitter::INTERPRETER, result.error);
174 return result;
175 }
176
177 LILA_DEBUG(Emitter::INTERPRETER, "Evaluating code...");
178
179 auto eval_result = m_impl->interpreter->ParseAndExecute(code);
180
181 if (!eval_result) {
182 result.success = true;
183 LILA_DEBUG(Emitter::INTERPRETER, "Code evaluation succeeded");
184 } else {
185 result.success = false;
186 result.error = "Execution failed: " + llvm::toString(std::move(eval_result));
187 LILA_ERROR(Emitter::INTERPRETER, result.error);
188 }
189
190 m_impl->eval_counter++;
191 return result;
192}

210void* ClangInterpreter::get_symbol_address(const std::string& name)
211{
212 if (!m_impl->interpreter) {
213 LILA_WARN(Emitter::INTERPRETER, "Cannot get symbol: interpreter not initialized");
214 return nullptr;
215 }
216
217 auto symbol = m_impl->interpreter->getSymbolAddress(name);
218 if (symbol) {
219 LILA_DEBUG(Emitter::INTERPRETER, std::string("Found symbol: ") + name);
220 return reinterpret_cast<void*>(symbol->getValue());
221 }
222
223 LILA_WARN(Emitter::INTERPRETER, std::string("Symbol not found: ") + name);
224 return nullptr;
225}
