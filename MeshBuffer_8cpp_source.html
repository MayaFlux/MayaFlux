108void MeshBuffer::setup_rendering(const RenderConfig& config)
109{
110 if (!config.vertex_shader.empty()) {
111 m_render_config.vertex_shader = config.vertex_shader;
112 }
113 if (!config.fragment_shader.empty()) {
114 m_render_config.fragment_shader = config.fragment_shader;
115 }
116 if (!config.default_texture_binding.empty()) {
117 m_render_config.default_texture_binding = config.default_texture_binding;
118 }
119
120 if (!m_render_config.default_texture_binding.empty()
121 && m_diffuse_binding != m_render_config.default_texture_binding) {
122 m_diffuse_binding = m_render_config.default_texture_binding;
123 }
124
125 m_render_config.target_window = config.target_window;
126 m_render_config.topology = Portal::Graphics::PrimitiveTopology::TRIANGLE_LIST;
127
128 if (!config.additional_textures.empty()) {
129 for (const auto& [name, texture] : config.additional_textures) {
130 m_render_config.additional_textures.emplace_back(name, texture);
131 }
132 }
133
134 const bool textured = m_diffuse_texture != nullptr
135 || !m_render_config.default_texture_binding.empty();
136
137 if (textured && m_render_config.default_texture_binding.empty()) {
138 m_render_config.default_texture_binding = m_diffuse_binding;
139 }
140
141 if (m_render_config.fragment_shader.empty()) {
142 m_render_config.fragment_shader = textured
143 ? "mesh_textured.frag.spv"
144 : "triangle.frag.spv";
145 }
146
147 ShaderConfig sc { m_render_config.vertex_shader };
148
149 if (textured && !m_render_config.default_texture_binding.empty()) {
150 sc.bindings[m_render_config.default_texture_binding] = ShaderBinding(
151 0, 1, vk::DescriptorType::eCombinedImageSampler);
152 }
153
154 const uint32_t additional_start = textured ? 2 : 1;
155 uint32_t binding_index = additional_start;
156 for (const auto& [name, _] : m_render_config.additional_textures) {
157 sc.bindings[name] = ShaderBinding(
158 0, binding_index++, vk::DescriptorType::eCombinedImageSampler);
159 }
160
161 apply_render_config(m_render_config, sc);
162
163 if (m_diffuse_texture && !m_render_config.default_texture_binding.empty()) {
164 m_render_processor->bind_texture(
165 m_render_config.default_texture_binding,
166 m_diffuse_texture);
167 }
168
169 for (const auto& [name, texture] : m_render_config.additional_textures) {
170 m_render_processor->bind_texture(name, texture);
171 }
172
173 get_processing_chain()->add_final_processor(
174 m_render_processor,
175 shared_from_this());
176
177 MF_INFO(Journal::Component::Buffers, Journal::Context::Init,
178 "MeshBuffer::setup_rendering: vert={} frag={} textured={}",
179 m_render_config.vertex_shader,
180 m_render_config.fragment_shader,
181 textured);
182}

211void MeshBuffer::set_vertex_data(std::span<const uint8_t> bytes)
212{
213 if (m_mesh_data.layout.stride_bytes > 0
214 && bytes.size() % m_mesh_data.layout.stride_bytes != 0) {
215 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
216 "MeshBuffer::set_vertex_data: byte count {} not a multiple of stride {}",
217 bytes.size(), m_mesh_data.layout.stride_bytes);
218 return;
219 }
220
221 m_mesh_data.vertex_variant = std::vector<uint8_t>(bytes.begin(), bytes.end());
222 m_mesh_data.layout.vertex_count = m_mesh_data.layout.stride_bytes > 0
223 ? static_cast<uint32_t>(bytes.size() / m_mesh_data.layout.stride_bytes)
224 : 0;
225
226 m_vertices_dirty.store(true, std::memory_order_release);
227}
