17const Kakshya::VertexLayout* RenderProcessor::get_or_cache_vertex_layout(
18 std::unordered_map<std::shared_ptr<VKBuffer>, RenderProcessor::VertexInfo>& buffer_info,
19 const std::shared_ptr<VKBuffer>& buffer)
20{
21 auto info_it = buffer_info.find(buffer);
22 if (info_it == buffer_info.end()) {
23 if (buffer->has_vertex_layout()) {
24 auto vertex_layout = buffer->get_vertex_layout();
25 if (vertex_layout.has_value()) {
26 buffer_info[buffer] = { .semantic_layout = vertex_layout.value(), .use_reflection = false };
27 info_it = buffer_info.find(buffer);
28 }
29 }
30 if (info_it == buffer_info.end()) {
31 return nullptr;
32 }
33 }
34 return &info_it->second.semantic_layout;
35}

125void RenderProcessor::initialize_pipeline(const std::shared_ptr<VKBuffer>& buffer)
126{
127 if (m_shader_id == Portal::Graphics::INVALID_SHADER) {
128 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
129 "Vertex shader not loaded");
130 return;
131 }
132
133 if (m_fragment_shader_id == Portal::Graphics::INVALID_SHADER) {
134 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
135 "Fragment shader not loaded");
136 return;
137 }
138
139 if (!m_target_window) {
140 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
141 "Target window not set");
142 return;
143 }
144
145 auto& flow = Portal::Graphics::get_render_flow();
146
147 Portal::Graphics::get_render_flow().register_window_for_rendering(m_target_window);
148
149 Portal::Graphics::RenderPipelineConfig pipeline_config;
150 pipeline_config.vertex_shader = m_shader_id;
151 pipeline_config.fragment_shader = m_fragment_shader_id;
152 pipeline_config.geometry_shader = m_geometry_shader_id;
153 pipeline_config.tess_control_shader = m_tess_control_shader_id;
154 pipeline_config.tess_eval_shader = m_tess_eval_shader_id;
155
156 pipeline_config.topology = m_primitive_topology;
157 pipeline_config.rasterization.polygon_mode = m_polygon_mode;
158 pipeline_config.rasterization.cull_mode = m_cull_mode;
159
160 pipeline_config.blend_attachments.emplace_back();
161
162 if (m_buffer_info.find(buffer) == m_buffer_info.end()) {
163 if (buffer->has_vertex_layout()) {
164 auto vertex_layout = buffer->get_vertex_layout();
165 if (vertex_layout.has_value()) {
166 m_buffer_info[buffer] = {
167 .semantic_layout = vertex_layout.value(),
168 .use_reflection = false
169 };
170 }
171 }
172 }
173
174 const Kakshya::VertexLayout* local_layout = get_or_cache_vertex_layout(m_buffer_info, buffer);
175 if (!local_layout) {
176 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
177 "initialize_pipeline: layout not yet available, deferring");
178 return;
179 }
180
181 pipeline_config.semantic_vertex_layout = *local_layout;
182 pipeline_config.use_vertex_shader_reflection = m_buffer_info[buffer].use_reflection;
183
184 const auto& staging = buffer->get_pipeline_context().push_constant_staging;
185 if (!staging.empty()) {
186 pipeline_config.push_constant_size = staging.size();
187 } else {
188 pipeline_config.push_constant_size = std::max(m_config.push_constant_size, m_push_constant_data.size());
189 }
190
191 auto& descriptor_bindings = buffer->get_pipeline_context().descriptor_buffer_bindings;
192 std::map<std::pair<uint32_t, uint32_t>, Portal::Graphics::DescriptorBindingInfo> unified_bindings;
193
194 for (const auto& binding : descriptor_bindings) {
195 unified_bindings[{ binding.set, binding.binding }] = binding;
196 }
197
198 for (const auto& [name, binding] : m_config.bindings) {
199 auto key = std::make_pair(binding.set, binding.binding);
200 if (unified_bindings.find(key) == unified_bindings.end()) {
201 unified_bindings[key] = Portal::Graphics::DescriptorBindingInfo {
202 .set = binding.set,
203 .binding = binding.binding,
204 .type = binding.type,
205 .buffer_info = {},
206 .name = name
207 };
208 }
209 }
210
211 std::map<uint32_t, std::vector<Portal::Graphics::DescriptorBindingInfo>> bindings_by_set;
212 for (const auto& [key, binding] : unified_bindings) {
213 bindings_by_set[binding.set].push_back(binding);
214 }
215
216 for (const auto& [set_index, set_bindings] : bindings_by_set) {
217 pipeline_config.descriptor_sets.push_back(set_bindings);
218 }
219
220 vk::Format swapchain_format = static_cast<vk::Format>(
221 m_display_service->get_swapchain_format(m_target_window));
222
223 m_pipeline_id = flow.create_pipeline(pipeline_config, { swapchain_format });
224
225 if (m_pipeline_id == Portal::Graphics::INVALID_RENDER_PIPELINE) {
226 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
227 "Failed to create render pipeline");
228 return;
229 }
230
231 m_needs_descriptor_rebuild = !pipeline_config.descriptor_sets.empty() && m_descriptor_set_ids.empty();
232 m_needs_pipeline_rebuild = false;
233
234 on_pipeline_created(m_pipeline_id);
235}

237void RenderProcessor::initialize_descriptors(const std::shared_ptr<VKBuffer>& buffer)
238{
239 if (m_pipeline_id == Portal::Graphics::INVALID_RENDER_PIPELINE) {
240 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
241 "Cannot allocate descriptor sets without pipeline");
242 return;
243 }
244
245 on_before_descriptors_create();
246
247 auto& flow = Portal::Graphics::get_render_flow();
248
249 m_descriptor_set_ids = flow.allocate_pipeline_descriptors(m_pipeline_id);
250
251 if (m_descriptor_set_ids.empty()) {
252 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
253 "Failed to allocate descriptor sets for pipeline");
254 return;
255 }
256
257 for (const auto& [binding, tex_binding] : m_texture_bindings) {
258 auto config_it = std::ranges::find_if(m_config.bindings,
259 [binding](const auto& pair) {
260 return pair.second.binding == binding;
261 });
262
263 if (config_it == m_config.bindings.end()) {
264 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
265 "No config for binding {}", binding);
266 continue;
267 }
268 uint32_t set_index = config_it->second.set;
269
270 if (set_index >= m_descriptor_set_ids.size()) {
271 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
272 "Descriptor set index {} out of range", binding);
273 continue;
274 }
275
276 auto& foundry = Portal::Graphics::get_shader_foundry();
277 foundry.update_descriptor_image(
278 m_descriptor_set_ids[set_index],
279 config_it->second.binding,
280 tex_binding.texture->get_image_view(),
281 tex_binding.sampler,
282 vk::ImageLayout::eShaderReadOnlyOptimal);
283 }
284
285 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
286 "Allocated {} descriptor sets and updated {} texture bindings",
287 m_descriptor_set_ids.size(), m_texture_bindings.size());
288
289 update_descriptors(buffer);
290 on_descriptors_created();
291}

324void RenderProcessor::execute_shader(const std::shared_ptr<VKBuffer>& buffer)
325{
326 if (m_buffer_info.find(buffer) == m_buffer_info.end()) {
327 if (buffer->has_vertex_layout()) {
328 auto vertex_layout = buffer->get_vertex_layout();
329 if (vertex_layout.has_value()) {
330 m_buffer_info[buffer] = {
331 .semantic_layout = vertex_layout.value(),
332 .use_reflection = false
333 };
334 }
335 }
336 }
337
338 if (!m_target_window->is_graphics_registered()) {
339 return;
340 }
341
342 if (m_pipeline_id == Portal::Graphics::INVALID_RENDER_PIPELINE) {
343 return;
344 }
345
346 const Kakshya::VertexLayout* local_layout = get_or_cache_vertex_layout(m_buffer_info, buffer);
347 if (!local_layout) {
348 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
349 "VKBuffer has no vertex layout set. Use buffer->set_vertex_layout()");
350 return;
351 }
352
353 buffer->set_pipeline_window(m_pipeline_id, m_target_window);
354
355 auto& foundry = Portal::Graphics::get_shader_foundry();
356 auto& flow = Portal::Graphics::get_render_flow();
357
358 vk::Format color_format = static_cast<vk::Format>(
359 m_display_service->get_swapchain_format(m_target_window));
360
361 auto cmd_id = foundry.begin_secondary_commands(color_format);
362 auto cmd = foundry.get_command_buffer(cmd_id);
363
364 uint32_t width = 0, height = 0;
365 m_display_service->get_swapchain_extent(m_target_window, width, height);
366
367 if (width > 0 && height > 0) {
368 auto cmd = foundry.get_command_buffer(cmd_id);
369
370 vk::Viewport viewport {
371 0.0F,
372 static_cast<float>(height),
373 static_cast<float>(width),
374 -static_cast<float>(height),
375 0.0F,
376 1.0F
377 };
378 cmd.setViewport(0, 1, &viewport);
379
380 vk::Rect2D scissor { { 0, 0 }, { width, height } };
381 cmd.setScissor(0, 1, &scissor);
382 }
383
384 flow.bind_pipeline(cmd_id, m_pipeline_id);
385
386 auto& descriptor_bindings = buffer->get_pipeline_context().descriptor_buffer_bindings;
387 if (!descriptor_bindings.empty()) {
388 for (const auto& binding : descriptor_bindings) {
389 if (binding.set >= m_descriptor_set_ids.size()) {
390 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
391 "Descriptor set index {} out of range", binding.set);
392 continue;
393 }
394
395 foundry.update_descriptor_buffer(
396 m_descriptor_set_ids[binding.set],
397 binding.binding,
398 binding.type,
399 binding.buffer_info.buffer,
400 binding.buffer_info.offset,
401 binding.buffer_info.range);
402 }
403 }
404
405 if (!m_descriptor_set_ids.empty()) {
406 flow.bind_descriptor_sets(cmd_id, m_pipeline_id, m_descriptor_set_ids);
407 }
408
409 const auto& staging = buffer->get_pipeline_context();
410 if (!staging.push_constant_staging.empty()) {
411 flow.push_constants(
412 cmd_id,
413 m_pipeline_id,
414 staging.push_constant_staging.data(),
415 staging.push_constant_staging.size());
416 } else if (!m_push_constant_data.empty()) {
417 flow.push_constants(
418 cmd_id,
419 m_pipeline_id,
420 m_push_constant_data.data(),
421 m_push_constant_data.size());
422 }
423
424 on_before_execute(cmd_id, buffer);
425
426 flow.bind_vertex_buffers(cmd_id, { buffer });
427
428 uint32_t draw_count = 0;
429 if (m_vertex_count > 0) {
430 draw_count = m_vertex_count;
431 } else {
432 auto current_layout = buffer->get_vertex_layout();
433 if (!current_layout.has_value() || current_layout->vertex_count == 0) {
434 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
435 "Vertex layout has zero vertices, skipping draw");
436 return;
437 }
438 draw_count = current_layout->vertex_count;
439 }
440
441 flow.draw(cmd_id, draw_count, 1, m_first_vertex, 0);
442
443 foundry.end_commands(cmd_id);
444
445 buffer->set_pipeline_command(m_pipeline_id, cmd_id);
446 m_target_window->track_frame_command(cmd_id);
447
448 MF_RT_TRACE(Journal::Component::Buffers, Journal::Context::BufferProcessing,
449 "Recorded secondary command buffer {} for window '{}'",
450 cmd_id, m_target_window->get_create_info().title);
451}

478void RenderProcessor::cleanup()
479{

481 auto& flow = Portal::Graphics::get_render_flow();
482
483 if (m_pipeline_id != Portal::Graphics::INVALID_RENDER_PIPELINE) {

495 m_tess_control_shader_id = Portal::Graphics::INVALID_SHADER;
496 }
497
498 if (m_tess_eval_shader_id != Portal::Graphics::INVALID_SHADER) {
499 foundry.destroy_shader(m_tess_eval_shader_id);
500 m_tess_eval_shader_id = Portal::Graphics::INVALID_SHADER;
501 }

504 foundry.destroy_shader(m_fragment_shader_id);
505 m_fragment_shader_id = Portal::Graphics::INVALID_SHADER;
506 }
507
508 if (m_target_window) {
509 flow.unregister_window(m_target_window);

513 ShaderProcessor::cleanup();

515 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
