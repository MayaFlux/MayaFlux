58void RenderProcessor::initialize_pipeline(const std::shared_ptr<Buffer>& buffer)
59{
60 if (m_shader_id == Portal::Graphics::INVALID_SHADER) {
61 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
62 "Vertex shader not loaded");
63 return;
64 }
65
66 if (m_fragment_shader_id == Portal::Graphics::INVALID_SHADER) {
67 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
68 "Fragment shader not loaded");
69 return;
70 }
71
72 auto& flow = Portal::Graphics::get_render_flow();
73
74 if (!m_target_window) {
75 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
76 "Target window not set");
77 return;
78 }
79
80 // vk::Format swapchain_format = vk::Format { m_display_service->get_swapchain_format(m_target_window) };
81 // m_render_pass_id = flow.create_simple_render_pass(swapchain_format);
82
83 if (m_render_pass_id == Portal::Graphics::INVALID_RENDER_PASS) {
84 vk::Format swapchain_format = vk::Format { m_display_service->get_swapchain_format(m_target_window) };
85 m_render_pass_id = flow.create_simple_render_pass(swapchain_format);
86
87 if (m_render_pass_id == Portal::Graphics::INVALID_RENDER_PASS) {
88 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
89 "Render pass not set");
90 return;
91 }
92 }
93
94 Portal::Graphics::get_render_flow().register_window_for_rendering(m_target_window, m_render_pass_id);
95
96 if (!buffer) {
97 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
98 "Cannot initialize pipeline: buffer is null");
99 return;
100 }
101 auto vk_buffer = std::dynamic_pointer_cast<VKBuffer>(buffer);
102
103 Portal::Graphics::RenderPipelineConfig pipeline_config;
104 pipeline_config.vertex_shader = m_shader_id;
105 pipeline_config.fragment_shader = m_fragment_shader_id;
106 pipeline_config.geometry_shader = m_geometry_shader_id;
107 pipeline_config.tess_control_shader = m_tess_control_shader_id;
108 pipeline_config.tess_eval_shader = m_tess_eval_shader_id;
109 pipeline_config.render_pass = m_render_pass_id;
110
111 pipeline_config.topology = m_primitive_topology;
112 pipeline_config.rasterization.polygon_mode = m_polygon_mode;
113 pipeline_config.rasterization.cull_mode = m_cull_mode;
114
115 pipeline_config.blend_attachments.emplace_back();
116
117 auto it = m_buffer_info.find(vk_buffer);
118 if (it != m_buffer_info.end()) {
119 const auto& vertex_info = it->second;
120 pipeline_config.semantic_vertex_layout = vertex_info.semantic_layout;
121 pipeline_config.use_vertex_shader_reflection = vertex_info.use_reflection;
122 }
123
124 m_render_pipeline_id = flow.create_pipeline(pipeline_config);
125
126 if (m_render_pipeline_id == Portal::Graphics::INVALID_RENDER_PIPELINE) {
127 MF_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
128 "Failed to create render pipeline");
129 return;
130 }
131
132 m_needs_pipeline_rebuild = false;
133}

135void RenderProcessor::processing_function(std::shared_ptr<Buffer> buffer)
136{
137 auto vk_buffer = std::dynamic_pointer_cast<VKBuffer>(buffer);
138 if (!vk_buffer || !m_target_window) {
139 return;
140 }
141
142 if (m_buffer_info.find(vk_buffer) == m_buffer_info.end()) {
143 if (vk_buffer->has_vertex_layout()) {
144 auto vertex_layout = vk_buffer->get_vertex_layout();
145 if (vertex_layout.has_value()) {
146 m_buffer_info[vk_buffer] = {
147 .semantic_layout = vertex_layout.value(),
148 .use_reflection = false
149 };
150 }
151 }
152 }
153
154 if (m_needs_pipeline_rebuild) {
155 initialize_pipeline(buffer);
156 }
157
158 if (m_render_pipeline_id == Portal::Graphics::INVALID_RENDER_PIPELINE) {
159 return;
160 }
161
162 auto vertex_layout = vk_buffer->get_vertex_layout();
163 if (!vertex_layout.has_value()) {
164 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
165 "VKBuffer has no vertex layout set. Use buffer->set_vertex_layout()");
166 return;
167 }
168
169 if (vertex_layout->vertex_count == 0) {
170 MF_RT_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
171 "Vertex layout has zero vertices, skipping draw");
172 return;
173 }
174
175 if (vertex_layout->attributes.empty()) {
176 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
177 "Vertex layout has no attributes");
178 return;
179 }
180
181 vk_buffer->set_pipeline_window(m_render_pipeline_id, m_target_window);
182
183 auto& foundry = Portal::Graphics::get_shader_foundry();
184 auto& flow = Portal::Graphics::get_render_flow();
185
186 auto cmd_id = foundry.begin_commands(Portal::Graphics::ShaderFoundry::CommandBufferType::GRAPHICS);
187
188 flow.begin_render_pass(cmd_id, m_target_window);
189
190 uint32_t width = 0, height = 0;
191 m_display_service->get_swapchain_extent(m_target_window, width, height);
192
193 if (width > 0 && height > 0) {
194 auto cmd = foundry.get_command_buffer(cmd_id);
195
196 vk::Viewport viewport { 0.0F, 0.0F, static_cast<float>(width), static_cast<float>(height), 0.0F, 1.0F };
197 cmd.setViewport(0, 1, &viewport);
198
199 vk::Rect2D scissor { { 0, 0 }, { width, height } };
200 cmd.setScissor(0, 1, &scissor);
201 }
202
203 flow.bind_pipeline(cmd_id, m_render_pipeline_id);
204
205 flow.bind_vertex_buffers(cmd_id, { vk_buffer });
206
207 flow.draw(cmd_id, vertex_layout->vertex_count);
208
209 flow.end_render_pass(cmd_id);
210
211 vk_buffer->set_pipeline_command(m_render_pipeline_id, cmd_id);
212}

239void RenderProcessor::cleanup()
240{
241 auto& foundry = Portal::Graphics::get_shader_foundry();
242 auto& flow = Portal::Graphics::get_render_flow();
243
244 if (m_render_pipeline_id != Portal::Graphics::INVALID_RENDER_PIPELINE) {
245 flow.destroy_pipeline(m_render_pipeline_id);
246 m_render_pipeline_id = Portal::Graphics::INVALID_RENDER_PIPELINE;
247 }
248
249 if (m_render_pass_id != Portal::Graphics::INVALID_RENDER_PASS) {
250 flow.destroy_render_pass(m_render_pass_id);
251 m_render_pass_id = Portal::Graphics::INVALID_RENDER_PASS;
252 }
253
254 if (m_geometry_shader_id != Portal::Graphics::INVALID_SHADER) {
255 foundry.destroy_shader(m_geometry_shader_id);
256 m_geometry_shader_id = Portal::Graphics::INVALID_SHADER;
257 }
258
259 if (m_tess_control_shader_id != Portal::Graphics::INVALID_SHADER) {
260 foundry.destroy_shader(m_tess_control_shader_id);
261 m_tess_control_shader_id = Portal::Graphics::INVALID_SHADER;
262 }
263
264 if (m_tess_eval_shader_id != Portal::Graphics::INVALID_SHADER) {
265 foundry.destroy_shader(m_tess_eval_shader_id);
266 m_tess_eval_shader_id = Portal::Graphics::INVALID_SHADER;
267 }
268
269 if (m_fragment_shader_id != Portal::Graphics::INVALID_SHADER) {
270 foundry.destroy_shader(m_fragment_shader_id);
271 m_fragment_shader_id = Portal::Graphics::INVALID_SHADER;
272 }
273
274 if (m_target_window) {
275 flow.unregister_window(m_target_window);
276 m_target_window.reset();
277 }
278
279 ShaderProcessor::cleanup();
280
281 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
282 "RenderProcessor cleanup complete");
283}
