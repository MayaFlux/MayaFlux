12/**
13 * @file VisionExecutor.hpp
14 * @brief Dispatch engine for VisionSequence execution.
15 *
16 * VisionExecutor::run() is the single entry point for executing a
17 * VisionSequence. It maintains per-frame state (previous gray frame for
18 * optical flow) and returns a VisionResult carrying both the final pixel
19 * image and any structured outputs produced by the terminal step.
20 *
21 * Scratch storage is owned by the executor as DataVariant slots holding
22 * vector<float>. Steps ping-pong between slots via Eigen::Map<ArrayXf>
23 * views, accumulating zero heap allocation in steady state at a fixed
24 * resolution. Gaussian kernels are cached by sigma across frames.
25 *
26 * VisionResult::pixel_image is a DataVariant (vector<float>) moved out of
27 * a scratch slot. Callers read it via EigenAccess::view<Eigen::VectorXf>()
28 * for zero-copy Eigen access, or as_span() for raw float access.
29 */
30

41/**
42 * @brief Result of executing a VisionSequence on one frame.
43 *
44 * pixel_image holds the final normalised float pixel buffer as a DataVariant
45 * (active alternative: vector<float>). Empty when the terminal step produces
46 * only structured output.
47 *
48 * Callers access pixel data via:
49 * EigenAccess(result.pixel_image).view<Eigen::VectorXf>() -- zero-copy Eigen map
50 * std::get<std::vector<float>>(result.pixel_image) -- direct vector access
51 *
52 * w and h are the dimensions of pixel_image. Both are 0 when pixel_image is empty.
53 */
54struct VisionResult {
55 Kakshya::DataVariant pixel_image { std::vector<float> {} };
56 StructuredOutput structured { std::monostate {} };
57 std::vector<SnapshotEntry> snapshots;
58 uint32_t w { 0 };
59 uint32_t h { 0 };
60
61 /**
62 * @brief Zero-copy float span into pixel_image storage.
63 * @return Empty span if pixel_image is not vector<float> or is empty.
64 */
65 [[nodiscard]] std::span<const float> as_span() const noexcept
66 {
67 const auto* v = std::get_if<std::vector<float>>(&pixel_image);
68 if (!v || v->empty())
69 return {};
70 return { v->data(), v->size() };
71 }
72};

74/**
75 * @class VisionExecutor
76 * @brief Stateful executor for a VisionSequence.
77 *
78 * Owns a pool of DataVariant scratch slots (each holding vector<float>) sized
79 * to the working resolution. Steps read via Eigen::Map<const ArrayXf> and
80 * write via Eigen::Map<ArrayXf> into the next slot, then swap indices.
81 * No heap allocation occurs in steady state after the first frame at a given
82 * resolution.
83 *
84 * Gaussian kernels are cached by sigma (keyed on bit-exact float) and reused
85 * across frames and across the three structure tensor smoothing passes in Harris.
86 *
87 * One executor instance per pipeline. Not thread-safe.
88 */

101 * @return VisionResult with pixel_image (DataVariant) and/or structured output.

117 // =========================================================================
118 // Scratch pool
119 //
120 // Each slot is a DataVariant holding vector<float>. Slots are accessed via
121 // slot_vec(i) for direct vector reference and slot_map(i) / slot_map_mut(i)
122 // for zero-copy Eigen views. The working resolution is tracked to detect
123 // geometry changes and resize all slots in one pass.
124 //
125 // Slot assignment (fixed):
126 // 0 current working buffer (ping)
127 // 1 next working buffer (pong)
128 // 2 filter horizontal pass tmp
129 // 3 harris: dx
130 // 4 harris: dy
131 // 5 harris: ixx
132 // 6 harris: iyy
133 // 7 harris: ixy
134 // 8 harris: sxx (smoothed)
135 // 9 harris: syy
136 // 10 harris: sxy
137 // =========================================================================
