10/**
11 * @file Geometry.hpp
12 * @brief Illustrative geometry functions for common Mapped use cases.
13 *
14 * These are starting points for reading and understanding the GeometryFn
15 * contract, not the primary or idiomatic way to use Mapped in MayaFlux.
16 *
17 * The idiomatic use is a custom geometry function that writes from whatever
18 * data source makes sense: mouse pixel coordinates written directly as buffer
19 * data, microphone energy mapped to a spatial form, a node output driving
20 * vertex positions, a tendency field evaluated at runtime. The function
21 * receives a value and an output byte buffer — what it does with those is
22 * unconstrained.
23 *
24 * The helpers below demonstrate the pattern concretely. They are not
25 * privileged and carry no special status in the framework.
26 */

39// =============================================================================
40// Horizontal fader
41//
42// Value in [0, 1] moves a handle quad along a track quad.
43// Two quads: track (static) and handle (value-driven).
44// Both written as TRIANGLE_STRIP pairs into a single buffer.
45//
46// Track: full extent of bounds.
47// Handle: small square at x = bounds.min.x + value * bounds.width().
48//
49// Hit region follows the handle, updated on every sync().
50// =============================================================================
51
52/**
53 * @brief Geometry function for a horizontal fader in NDC space.
54 * @param bounds Full extent of the fader in NDC.
55 * @param handle_w Handle width in NDC units.
56 * @param track_color Track quad color.
57 * @param handle_color Handle quad color.
58 */
59[[nodiscard]] inline GeometryFn<float> horizontal_fader(
60 Kinesis::AABB2D bounds,
61 float handle_w,
62 glm::vec3 track_color = glm::vec3(0.3F),
63 glm::vec3 handle_color = glm::vec3(0.9F))
64{
65 return [bounds, handle_w, track_color, handle_color](
66 float v, std::vector<uint8_t>& out, Element& el) {
67 float x = bounds.min.x + v * (bounds.width() - handle_w);
68 float yt = bounds.min.y + bounds.height() * 0.35F;
69 float yb = bounds.min.y + bounds.height() * 0.65F;
70
71 using V = Nodes::LineVertex;
72 std::vector<V> verts;
73 verts.reserve(8);
74
75 // track — two triangles as TRIANGLE_STRIP
76 verts.push_back({ .position = { bounds.min.x, yt, 0 }, .color = track_color });
77 verts.push_back({ .position = { bounds.min.x, yb, 0 }, .color = track_color });
78 verts.push_back({ .position = { bounds.max.x, yt, 0 }, .color = track_color });
79 verts.push_back({ .position = { bounds.max.x, yb, 0 }, .color = track_color });
80
81 // handle — two triangles as TRIANGLE_STRIP
82 verts.push_back({ .position = { x, bounds.min.y, 0 }, .color = handle_color });
83 verts.push_back({ .position = { x, bounds.max.y, 0 }, .color = handle_color });
84 verts.push_back({ .position = { x + handle_w, bounds.min.y, 0 }, .color = handle_color });
85 verts.push_back({ .position = { x + handle_w, bounds.max.y, 0 }, .color = handle_color });
86
87 detail::write_verts(out, verts);
88
89 el.bounds_hint = Kinesis::AABB2D {
90 .min = glm::vec2(x, bounds.min.y),
91 .max = glm::vec2(x + handle_w, bounds.max.y)
92 };
93 el.contains = {};
94 };
95}

97// =============================================================================
98// Radial / arc
99//
100// Value in [0, 1] sweeps an indicator line around a center point.
101// angle_start and angle_end in radians, measured from +X, CCW.
102// Produces a LINE_LIST of two vertices: center to indicator tip.
103// =============================================================================
104
105/**
106 * @brief Geometry function for a radial indicator in NDC space.
107 * @param center Arc center in NDC.
108 * @param radius Arc radius in NDC units.
109 * @param angle_start Start angle in radians (value = 0).
110 * @param angle_end End angle in radians (value = 1).
111 * @param color Line color.
112 */
113[[nodiscard]] inline GeometryFn<float> radial(
114 glm::vec2 center,
115 float radius,
116 float angle_start,
117 float angle_end,
118 glm::vec3 color = glm::vec3(0.9F))
119{
120 return [center, radius, angle_start, angle_end, color](
121 float v, std::vector<uint8_t>& out, Element& el) {
122 float angle = angle_start + v * (angle_end - angle_start);
123 glm::vec2 tip = center + radius * glm::vec2(std::cos(angle), std::sin(angle));
124
125 using V = Nodes::LineVertex;
126 std::vector<V> verts = {
127 { .position = { center.x, center.y, 0 }, .color = color },
128 { .position = { tip.x, tip.y, 0 }, .color = color },
129 };
130
131 detail::write_verts(out, verts);
132
133 el.bounds_hint = Kinesis::AABB2D::from_ndc(center, glm::vec2(radius));
134 el.contains = Kinesis::circular_bounds(center, radius);
135 };
136}

139// 2D position picker

141// Value is glm::vec2 in [0,1]x[0,1], mapped to a point inside bounds.

145/**
146 * @brief Geometry function for a 2D position picker in NDC space.
147 * @param bounds Full extent of the pick area in NDC.
148 * @param color Point color.
149 * @param size Point size in pixels.
150 */
151[[nodiscard]] inline GeometryFn<glm::vec2> position_picker(
152 Kinesis::AABB2D bounds,
153 glm::vec3 color = glm::vec3(0.9F),
154 float size = 8.0F)
155{
156 return [bounds, color, size](
157 glm::vec2 v, std::vector<uint8_t>& out, Element& el) {
158 float x = bounds.min.x + v.x * bounds.width();
159 float y = bounds.min.y + v.y * bounds.height();
160
161 using V = Nodes::PointVertex;
162 std::vector<V> verts = {
163 { .position = { x, y, 0 }, .color = color, .size = size },
164 };
165
166 detail::write_verts(out, verts);
167
168 el.bounds_hint = bounds;
169 el.contains = Kinesis::polygon_bounds(std::span<const glm::vec2> {
170 std::array<glm::vec2, 4> {
171 bounds.min,
172 glm::vec2(bounds.max.x, bounds.min.y),
173 bounds.max,
174 glm::vec2(bounds.min.x, bounds.max.y) } });
175 };
176}
