163void update_routing_state(RoutingState& state)
164{
165 state.cycles_elapsed++;
166
167 double progress = 0.;
168 if (state.fade_cycles > 0) {
169 progress = std::min<double>(1.0, static_cast<double>(state.cycles_elapsed) / state.fade_cycles);
170 } else {
171 for (uint32_t ch = 0; ch < 32; ++ch) {
172 bool will_be_in = state.to_channels & (1 << ch);
173 state.amount[ch] = will_be_in ? 1.0 : 0.0;
174 }
175 state.phase = RoutingState::NONE;
176 return;
177 }
178
179 for (uint32_t ch = 0; ch < 32; ch++) {
180 bool was_in = state.from_channels & (1 << ch);
181 bool will_be_in = state.to_channels & (1 << ch);
182
183 if (was_in && will_be_in) {
184 state.amount[ch] = 1.0;
185 } else if (was_in && !will_be_in) {
186 state.amount[ch] = 1.0 - progress;
187 } else if (!was_in && will_be_in) {
188 state.amount[ch] = progress;
189 } else {
190 state.amount[ch] = 0.0;
191 }
192 }
193
194 if (state.cycles_elapsed >= state.fade_cycles) {
195 state.phase = RoutingState::COMPLETED;
196 }
197}
