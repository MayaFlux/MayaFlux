72void NodeGraphManager::process_token(ProcessingToken token, unsigned int num_samples)
73{
74 if (m_terminate_requested.load())
75 return;
76
77 auto roots = get_all_root_nodes(token);
78
79 if (auto it = m_token_processors.find(token); it != m_token_processors.end()) {
80 it->second(std::span<RootNode*>(roots.data(), roots.size()));
81 return;
82 }
83
84 if (!preprocess_networks(token)) {
85 return;
86 }
87
88 auto it = m_token_networks.find(token);
89 if (it != m_token_networks.end()) {
90 for (auto& network : it->second) {
91 if (!network || !network->is_enabled()) {
92 continue;
93 }
94
95 if (!network->is_processed_this_cycle()) {
96 network->mark_processing(true);
97 network->process_batch(num_samples);
98 network->mark_processing(false);
99 network->mark_processed(true);
100 }
101 }
102 }
103
104 postprocess_networks(token, std::nullopt);
105
106 if (token == ProcessingToken::AUDIO_RATE) {
107 for (auto* root : roots) {
108 root->process_batch(num_samples);
109 }
110 } else if (token == ProcessingToken::VISUAL_RATE) {
111 for (auto* root : roots) {
112 root->process_batch_frame(num_samples);
113 }
114 }

117std::vector<std::vector<double>> NodeGraphManager::process_audio_networks(ProcessingToken token, uint32_t num_samples, uint32_t channel)
118{
119 if (!preprocess_networks(token)) {
120 return {};
121 }
122
123 std::vector<std::vector<double>> all_network_outputs;
124
125 auto audio_it = m_audio_networks.find(token);
126 if (audio_it != m_audio_networks.end()) {
127 for (auto& network : audio_it->second) {
128 if (!network || !network->is_enabled()) {
129 continue;
130 }
131
132 if (!network->is_registered_on_channel(channel)) {
133 continue;
134 }
135
136 if (!network->is_processed_this_cycle()) {
137 network->mark_processing(true);
138 network->process_batch(num_samples);
139 network->mark_processing(false);
140 network->mark_processed(true);
141 }
142
143 const auto& net_buffer = network->get_audio_buffer();
144 if (net_buffer && network->get_output_mode() == Network::OutputMode::AUDIO_SINK) {
145 if (network->needs_channel_routing()) {
146 double scale = network->get_routing_state().amount[channel];
147 if (scale == 0.0)
148 continue;
149
150 if (scale == 1.0) {
151 all_network_outputs.push_back(*net_buffer);
152 } else {
153 std::vector<double> scaled_buffer = *net_buffer;
154 for (auto& sample : scaled_buffer)
155 sample *= scale;
156
157 all_network_outputs.push_back(std::move(scaled_buffer));
158 }
159 } else {
160 all_network_outputs.push_back(*net_buffer);
161 }
162 }
163 }
164 }
165
166 postprocess_networks(token, channel);
167 return all_network_outputs;
168}

252void NodeGraphManager::normalize_sample(double& sample, uint32_t num_nodes)
253{
254 if (num_nodes == 0)
255 return;
256
257 sample /= std::sqrt(static_cast<double>(num_nodes));
258
259 const double threshold = 0.95;
260 const double knee = 0.1;
261 const double abs_sample = std::abs(sample);
262
263 if (abs_sample > threshold) {
264 const double excess = abs_sample - threshold;
265 const double compressed_excess = std::tanh(excess / knee) * knee;
266 const double limited_abs = threshold + compressed_excess;
267 sample = std::copysign(limited_abs, sample);
268 }

681void NodeGraphManager::route_node_to_channels(
682 const std::shared_ptr<Node>& node,
683 const std::vector<uint32_t>& target_channels,
684 uint32_t fade_cycles,
685 ProcessingToken token)
686{
687 uint32_t current_channels = node->get_channel_mask();
688
689 uint32_t target_bitmask = 0;
690 for (auto ch : target_channels) {
691 target_bitmask |= (1 << ch);
692 }
693
694 uint32_t fade_blocks = (fade_cycles + m_registered_block_size - 1) / m_registered_block_size;
695 fade_blocks = std::max(1U, fade_blocks);
696
697 RoutingState state;
698 state.from_channels = current_channels;
699 state.to_channels = target_bitmask;
700 state.fade_cycles = fade_blocks;
701 state.phase = RoutingState::ACTIVE;
702
703 for (uint32_t ch = 0; ch < 32; ch++) {
704 state.amount[ch] = (current_channels & (1 << ch)) ? 1.0 : 0.0;
705 }
706
707 node->get_routing_state() = state;
708
709 for (auto ch : target_channels) {
710 if (!(current_channels & (1 << ch))) {
711 add_to_root(node, token, ch);
712 }
713 }
714}

716void NodeGraphManager::route_network_to_channels(
717 const std::shared_ptr<Network::NodeNetwork>& network,
718 const std::vector<uint32_t>& target_channels,
719 uint32_t fade_cycles,
720 ProcessingToken token)
721{
722 if (network->get_output_mode() != Network::OutputMode::AUDIO_SINK) {
723 MF_WARN(Journal::Component::Nodes,
724 Journal::Context::NodeProcessing,
725 "Attempted to route network that is not an audio sink. Operation ignored.");
726 return;
727 }
728
729 register_network_global(network);
730 network->set_enabled(true);
731
732 auto& networks = m_audio_networks[token];
733 if (std::ranges::find(networks, network) == networks.end()) {
734 networks.push_back(network);
735 }
736
737 uint32_t current_channels = network->get_channel_mask();
738
739 uint32_t target_bitmask = 0;
740 for (auto ch : target_channels) {
741 target_bitmask |= (1 << ch);
742 }
743
744 uint32_t combined_mask = current_channels | target_bitmask;
745 network->set_channel_mask(combined_mask);
746 for (auto ch : target_channels) {
747 network->add_channel_usage(ch);
748 ensure_root_exists(token, ch);
749 }
750
751 uint32_t fade_blocks = (fade_cycles + m_registered_block_size - 1) / m_registered_block_size;
752 fade_blocks = std::max(1u, fade_blocks);
753
754 RoutingState state;
755 state.from_channels = current_channels;
756 state.to_channels = target_bitmask;
757 state.fade_cycles = fade_blocks;
758 state.phase = RoutingState::ACTIVE;
759
760 for (uint32_t ch = 0; ch < 32; ch++) {
761 state.amount[ch] = (current_channels & (1 << ch)) ? 1.0 : 0.0;
762 }
763
764 network->get_routing_state() = state;
765}

769 std::vector<std::pair<std::shared_ptr<Node>, uint32_t>> nodes_to_remove;
770
771 for (const auto& [id, node] : m_Node_registry) {
772 if (!node->needs_channel_routing())
773 continue;
774
775 auto& state = node->get_routing_state();
776
777 if (state.phase == RoutingState::COMPLETED) {
778 for (uint32_t ch = 0; ch < 32; ch++) {
779 if ((state.from_channels & (1 << ch)) && !(state.to_channels & (1 << ch))) {
780 nodes_to_remove.emplace_back(node, ch);
781 }
782 }
783 state = RoutingState {};
784 }
785 }
786
787 for (auto& [node, channel] : nodes_to_remove) {
788 remove_from_root(node, token, channel);
789 }
790
791 std::vector<std::pair<std::shared_ptr<Network::NodeNetwork>, uint32_t>> networks_to_cleanup;
792
793 for (const auto& network : get_all_networks(token)) {
794 if (!network || !network->needs_channel_routing())
795 continue;
796
797 auto& state = network->get_routing_state();
798
799 if (state.phase == RoutingState::COMPLETED) {
800 network->set_channel_mask(state.to_channels);
801
802 for (uint32_t ch = 0; ch < 32; ch++) {
803 if ((state.from_channels & (1 << ch)) && !(state.to_channels & (1 << ch))) {
804 networks_to_cleanup.emplace_back(network, ch);
805 }
806 }
807 state = RoutingState {};
808 }
809 }
810
811 for (auto& [network, channel] : networks_to_cleanup) {
812 network->remove_channel_usage(channel);
813
814 if (network->get_channel_mask() == 0) {
815 auto& networks = m_audio_networks[token];
816 std::erase_if(networks, [&](const auto& n) { return n == network; });
817 unregister_network_global(network);
818 }
819 }
820}
