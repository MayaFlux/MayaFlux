54void NodeGraphManager::process_token(ProcessingToken token, unsigned int num_samples)
55{
56 auto roots = get_all_root_nodes(token);
57
58 if (auto it = m_token_processors.find(token); it != m_token_processors.end()) {
59 it->second(std::span<RootNode*>(roots.data(), roots.size()));
60 return;
61 }
62
63 if (token == ProcessingToken::AUDIO_RATE) {
64 for (auto* root : roots) {
65 root->process_batch(num_samples);
66 }
67 } else if (token == ProcessingToken::VISUAL_RATE) {
68 for (auto* root : roots) {

116void NodeGraphManager::normalize_sample(double& sample, uint32_t num_nodes)
117{
118 if (num_nodes == 0)
119 return;
120
121 sample /= std::sqrt(static_cast<double>(num_nodes));
122
123 const double threshold = 0.95;
124 const double knee = 0.1;
125 const double abs_sample = std::abs(sample);
126
127 if (abs_sample > threshold) {
128 const double excess = abs_sample - threshold;
129 const double compressed_excess = std::tanh(excess / knee) * knee;
130 const double limited_abs = threshold + compressed_excess;
131 sample = std::copysign(limited_abs, sample);
132 }
133}
