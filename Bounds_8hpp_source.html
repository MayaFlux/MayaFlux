12/**
13 * @struct AABB2D
14 * @brief Axis-aligned bounding rectangle in a 2D coordinate space.
15 *
16 * Used as a fast-reject hint in Portal::Forma::Element. All coordinates
17 * are in NDC by convention when used in Forma: x and y in [-1, +1],
18 * center origin, +Y up, matching ViewTransform.hpp and Windowing.hpp.
19 *
20 * Named constructors convert from other coordinate spaces into NDC.
21 */
22struct AABB2D {
23 glm::vec2 min;
24 glm::vec2 max;
25
26 [[nodiscard]] bool contains(glm::vec2 p) const noexcept
27 {
28 return p.x >= min.x && p.x <= max.x
29 && p.y >= min.y && p.y <= max.y;
30 }
31
32 [[nodiscard]] bool overlaps(const AABB2D& other) const noexcept
33 {
34 return min.x <= other.max.x && max.x >= other.min.x
35 && min.y <= other.max.y && max.y >= other.min.y;
36 }
37
38 [[nodiscard]] float width() const noexcept { return max.x - min.x; }
39 [[nodiscard]] float height() const noexcept { return max.y - min.y; }
40
41 [[nodiscard]] glm::vec2 center() const noexcept { return (min + max) * 0.5F; }
42
43 [[nodiscard]] AABB2D translated(glm::vec2 offset) const noexcept
44 {
45 return { .min = min + offset, .max = max + offset };
46 }
47
48 [[nodiscard]] AABB2D expanded(float margin) const noexcept
49 {
50 return { .min = min - glm::vec2(margin), .max = max + glm::vec2(margin) };
51 }

53 [[nodiscard]] static AABB2D from_ndc(glm::vec2 center, glm::vec2 half) noexcept

58 /**
59 * @brief Construct from pixel-space top-left rect, converted to NDC.
60 * @param x Left edge in pixels.
61 * @param y Top edge in pixels (top-left origin, +Y down).
62 * @param w Width in pixels.
63 * @param h Height in pixels.
64 * @param win_w Window width in pixels.
65 * @param win_h Window height in pixels.
66 */
67 [[nodiscard]] static AABB2D from_pixel(
68 float x, float y, float w, float h,
69 uint32_t win_w, uint32_t win_h) noexcept
70 {
71 auto nx = [&](float px) {
72 return (px / static_cast<float>(win_w)) * 2.0F - 1.0F;
73 };
74 auto ny = [&](float py) {
75 return 1.0F - (py / static_cast<float>(win_h)) * 2.0F;
76 };
77 return { .min = glm::vec2(nx(x), ny(y + h)), .max = glm::vec2(nx(x + w), ny(y)) };
78 }

80 /**
81 * @brief Construct from a TextureBuffer NDC position and scale.

83 * @param ndc_scale Extent as returned by TextureBuffer::get_scale().

86 glm::vec2 ndc_position, glm::vec2 ndc_scale) noexcept
87 {
88 glm::vec2 half = ndc_scale * 0.5F;
89 return { .min = ndc_position - half, .max = ndc_position + half };
90 }
91};

99/**
100 * @brief Containment test for a circle.
101 * @param center NDC center.
102 * @param radius Radius in NDC units.
103 */
104[[nodiscard]] inline std::function<bool(glm::vec2)>
105circular_bounds(glm::vec2 center, float radius) noexcept
106{
107 float r2 = radius * radius;
108 return [center, r2](glm::vec2 p) {
109 glm::vec2 d = p - center;
110 return d.x * d.x + d.y * d.y <= r2;
111 };
112}

114/**
115 * @brief Containment test for a convex or concave polygon.
116 *
117 * Uses the winding number algorithm, which handles both convex and
118 * non-convex polygons correctly. Vertices are in NDC, ordered either
119 * clockwise or counter-clockwise.
120 *
121 * @param vertices Polygon vertices. Copied into the closure.
122 */
123[[nodiscard]] inline std::function<bool(glm::vec2)>
124polygon_bounds(std::span<const glm::vec2> vertices)
125{
126 std::vector<glm::vec2> verts(vertices.begin(), vertices.end());
127 return [verts = std::move(verts)](glm::vec2 p) {
128 int winding = 0;
129 size_t n = verts.size();
130 for (size_t i = 0; i < n; ++i) {
131 glm::vec2 a = verts[i];
132 glm::vec2 b = verts[(i + 1) % n];
133 if (a.y <= p.y) {
134 if (b.y > p.y) {
135 float cross = (b.x - a.x) * (p.y - a.y)
136 - (b.y - a.y) * (p.x - a.x);
137 if (cross > 0.0F)
138 ++winding;
139 }
140 } else {
141 if (b.y <= p.y) {
142 float cross = (b.x - a.x) * (p.y - a.y)
143 - (b.y - a.y) * (p.x - a.x);
144 if (cross < 0.0F)
145 --winding;
146 }
147 }
148 }
149 return winding != 0;
150 };
151}

153/**
154 * @brief Containment test for a polyline with a uniform half-thickness.
155 *
156 * Covers open and closed curves equally. A point is inside if its
157 * perpendicular distance to any segment is within @p half_thickness.
158 * Suitable for stroke-based interactive regions and cable hit testing.
159 *
160 * @param points Ordered polyline vertices in NDC.
161 * @param half_thickness Distance threshold in NDC units.
162 */
163[[nodiscard]] inline std::function<bool(glm::vec2)>
164stroke_bounds(std::span<const glm::vec2> points, float half_thickness)
165{
166 std::vector<glm::vec2> pts(points.begin(), points.end());
167 float t2 = half_thickness * half_thickness;
168 return [pts = std::move(pts), t2](glm::vec2 p) {
169 for (size_t i = 0; i + 1 < pts.size(); ++i) {
170 glm::vec2 a = pts[i];
171 glm::vec2 b = pts[i + 1];
172 glm::vec2 ab = b - a;
173 float len2 = glm::dot(ab, ab);
174 float d2 = 0.0F;
175 if (len2 < 1e-12F) {
176 glm::vec2 ap = p - a;
177 d2 = glm::dot(ap, ap);
178 } else {
179 float t = glm::clamp(glm::dot(p - a, ab) / len2, 0.0F, 1.0F);
180 glm::vec2 proj = a + t * ab;
181 glm::vec2 diff = p - proj;
182 d2 = glm::dot(diff, diff);
183 }
184 if (d2 <= t2)
185 return true;
186 }
187 return false;
188 };
189}
