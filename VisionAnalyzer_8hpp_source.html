67/**
68 * @class VisionAnalyzer
69 * @brief UniversalAnalyzer that takes any pixel-bearing input, runs a
70 * VisionSequence via VisionExecutor, and surfaces structured results
71 * into the Yantra compute graph.
72 *
73 * InputType can be any ComputeData type carrying pixel data:
74 * - std::vector<DataVariant> (direct pixel variants)
75 * - std::shared_ptr<SignalSourceContainer> (TextureContainer, VideoStreamContainer, etc.)
76 * - Kakshya::Region / RegionGroup (pixel sub-region with container)
77 *
78 * extract_structured_native handles all unwrapping transparently. Width and
79 * height are read from SPATIAL_X / SPATIAL_Y dimension roles in the
80 * DataStructureInfo, or from metadata keys "width" and "height" as fallback.
81 *
82 * OutputType is std::vector<DataVariant> carrying VisionResult::pixel_image
83 * as vector<float>, matching the standard analyzer pipeline contract.
84 * Structured results (boxes, contours, keypoints, tracks, gradient,
85 * components) are stored via store_current_analysis and retrieved with
86 * get_vision_analysis() / extract_scalar_vision().
87 *
88 * @code
89 * auto va = std::make_shared<StandardVisionAnalyzer>(sequence);
90 * va->apply_operation(pixel_datum);
91 * auto analysis = va->get_vision_analysis();
92 * double n = extract_scalar_vision(analysis, "box_count");
93 * @endcode
94 */

171 output_type analyze_implementation(const input_type& input) override
172 {
173 try {
174 auto [native_spans, structure_info] = OperationHelper::extract_structured_native(
175 const_cast<input_type&>(input));
176
177 if (native_spans.empty()) {
178 error<std::runtime_error>(
179 Journal::Component::Yantra,
180 Journal::Context::ComputeMatrix,
181 std::source_location::current(),
182 "VisionAnalyzer: no pixel data in input");
183 }
184
185 uint32_t w = 0;
186 uint32_t h = 0;
187 for (const auto& dim : structure_info.dimensions) {
188 if (dim.role == Kakshya::DataDimension::Role::SPATIAL_X) {
189 w = static_cast<uint32_t>(dim.size);
190 } else if (dim.role == Kakshya::DataDimension::Role::SPATIAL_Y) {
191 h = static_cast<uint32_t>(dim.size);
192 }
193 }
194 if (w == 0)
195 w = Kakshya::get_metadata_value<uint32_t>(input.metadata, "width").value_or(0);
196 if (h == 0)
197 h = Kakshya::get_metadata_value<uint32_t>(input.metadata, "height").value_or(0);
198
199 if (w == 0 || h == 0) {
200 error<std::runtime_error>(
201 Journal::Component::Yantra,
202 Journal::Context::ComputeMatrix,
203 std::source_location::current(),
204 "VisionAnalyzer: width/height not resolvable from dimensions or metadata");
205 }
206
207 auto frame = std::visit([&](const auto& span) -> std::span<const float> {
208 using ElemT = typename std::decay_t<decltype(span)>::value_type;
209 if constexpr (std::is_same_v<ElemT, float>) {
210 return span;
211 } else {
212 using VecT = std::vector<ElemT>;
213 VecT tmp(span.begin(), span.end());
214 Kakshya::DataVariant var(std::move(tmp));
215 return Kakshya::as_normalised_float(var, m_float_storage);
216 }
217 },
218 native_spans[0]);
219
220 if (frame.empty()) {
221 error<std::runtime_error>(
222 Journal::Component::Yantra,
223 Journal::Context::ComputeMatrix,
224 std::source_location::current(),
225 "VisionAnalyzer: normalisation produced empty frame");
226 }
227
228 const Kinesis::Vision::VisionResult vr = m_executor.run(m_sequence, frame, w, h);
229
230 VisionAnalysis analysis;
231 if (const auto* v = std::get_if<std::vector<float>>(&vr.pixel_image))
232 analysis.pixel_image = *v;
233
234 analysis.w = vr.w;
235 analysis.h = vr.h;
236
237 std::visit([&](const auto& s) {
238 using T = std::decay_t<decltype(s)>;
239 if constexpr (std::is_same_v<T, Kinesis::Vision::GradientResult>) {
240 analysis.frame.gradient = s;
241 } else if constexpr (std::is_same_v<T, Kinesis::Vision::ComponentResult>) {
242 analysis.frame.components = s;
243 analysis.frame.boxes = s.boxes;
244 } else if constexpr (std::is_same_v<T, std::vector<Kinesis::Vision::Contour>>) {
245 analysis.frame.contours = s;
246 } else if constexpr (std::is_same_v<T, std::vector<Kinesis::Vision::Keypoint>>) {
247 analysis.frame.keypoints = s;
248 } else if constexpr (std::is_same_v<T, std::vector<Kinesis::Vision::TrackResult>>) {
249 analysis.frame.tracks = s;
250 }
251 },
252 vr.structured);
253
254 this->store_current_analysis(analysis);
255
256 return create_pipeline_output(input, vr, structure_info);
257
258 } catch (const std::exception& e) {
259 MF_ERROR(Journal::Component::Yantra, Journal::Context::ComputeMatrix,
260 "VisionAnalyzer: {}", e.what());
261 output_type err;
262 err.metadata = input.metadata;
263 err.metadata["error"] = std::string(e.what());
264 return err;
265 }
266 }

273 output_type create_pipeline_output(
274 const input_type& input,
275 const Kinesis::Vision::VisionResult& vr,
276 const DataStructureInfo& info)
277 {
278 std::vector<Kakshya::DataVariant> out_variants;
279 const auto* pix = std::get_if<std::vector<float>>(&vr.pixel_image);
280 if (pix && !pix->empty())
281 out_variants.emplace_back(*pix);
282
283 output_type out = this->convert_result(
284 std::vector<std::vector<double>> {}, info);
285 out.data = OperationHelper::reconstruct_from_double<OutputType>({}, info);
286
287 if constexpr (std::is_same_v<OutputType, std::vector<Kakshya::DataVariant>>) {
288 out.data = std::move(out_variants);
289 }
290
291 out.dimensions = {
292 Kakshya::DataDimension::spatial_2d(vr.w, vr.h)
293 };
294 out.modality = Kakshya::DataModality::IMAGE_2D;
295 out.metadata = input.metadata;
296 out.metadata["source_analyzer"] = std::string("VisionAnalyzer");
297 out.metadata["vision_w"] = vr.w;
298 out.metadata["vision_h"] = vr.h;
299 return out;
300 }
