17InspectResult Inspector::buffer(
18 const std::shared_ptr<Buffers::Buffer>& buf,
19 Layer& layer, Context& context,
20 const std::shared_ptr<Core::Window>& window,
21 LayoutCursor& cursor,
22 float x_min, float x_max, float row_h, int depth)
23{
24 const float ind = x_min + static_cast<float>(depth) * k_inspect_indent;
25 const std::string header_label = Reflect::short_dynamic_type_name(*buf);
26
27 std::vector<ValueSpec> values {
28 ValueSpec {
29 .label = "default_proc",
30 .reader = [buf] {
31 auto p = buf->get_default_processor();
32 return p ? std::string(Reflect::short_dynamic_type_name(*p)) : "-";
33 },
34 },
35 };
36
37 auto chain = buf->get_processing_chain();
38 if (chain) {
39 auto pre = chain->get_preprocessor(buf);
40 values.push_back(ValueSpec {
41 .label = "pre",
42 .reader = [pre] {
43 return pre ? std::string(Reflect::short_dynamic_type_name(*pre)) : "-";
44 },
45 });
46
47 const auto& processors = chain->get_processors(buf);
48 for (size_t i = 0; i < processors.size(); ++i) {
49 auto p = processors[i];
50 values.push_back(ValueSpec {
51 .label = "chain[" + std::to_string(i) + "]",
52 .reader = [p] {
53 return p ? std::string(Reflect::short_dynamic_type_name(*p)) : "-";
54 },
55 });
56 }
57
58 auto post = chain->get_postprocessor(buf);
59 values.push_back(ValueSpec {
60 .label = "post",
61 .reader = [post] {
62 return post ? std::string(Reflect::short_dynamic_type_name(*post)) : "-";
63 },
64 });
65
66 auto final_p = chain->get_final_processor(buf);
67 values.push_back(ValueSpec {
68 .label = "final",
69 .reader = [final_p] {
70 return final_p ? std::string(Reflect::short_dynamic_type_name(*final_p)) : "-";
71 },
72 });
73 }
74
75 auto group = make_value_group(
76 header_label, values,
77 layer, context, window, cursor,
78 ind, x_max, row_h, false);
79
80 InspectResult result;
81 result.group = std::move(group);
82 return result;
83}

101

103 ValueSpec {
104 .label = "samples",
105 .reader = [root] { return std::to_string(root->get_num_samples()); },
106 },
107 ValueSpec {
108 .label = "children",
109 .reader = [root] { return std::to_string(root->get_num_children()); },
110 },
111 };
112

115 layer, context, window, cursor,
116 ind, x_max, row_h, false);
117
118 InspectResult result;
119 result.group = std::move(group);
120
121 auto root_result = buffer(
122 root, layer, context, window, cursor,
123 x_min, x_max, row_h, depth + 1);
124 layer.relate(result.group.header.header_id, root_result.group.header.header_id);
125 result.children.push_back(std::move(root_result));
126
127 for (const auto& child : children) {
128 auto child_result = buffer(
129 child, layer, context, window, cursor,

237InspectResult Inspector::buffer_manager(
238 Layer& layer, Context& context,
239 const std::shared_ptr<Core::Window>& window,
240 LayoutCursor& cursor,
241 float x_min, float x_max, float row_h)
242{
243 const auto tokens = m_bm.get_active_tokens();
244 const uint32_t in_count = m_bm.get_num_input_channels();
245
246 const std::string header_label = "BufferManager";
247
248 auto& bm = m_bm;
249 std::vector<ValueSpec> values {
250 ValueSpec {
251 .label = "tokens",
252 .reader = [&bm] { return std::to_string(bm.get_active_tokens().size()); },
253 },
254 ValueSpec {
255 .label = "inputs",
256 .reader = [&bm] { return std::to_string(bm.get_num_input_channels()); },
257 },
258 };
259
260 auto group = make_value_group(
261 header_label, values,
262 layer, context, window, cursor,
263 x_min, x_max, row_h, false);
264
265 InspectResult result;
266 result.group = std::move(group);
267
268 for (const auto tok : tokens) {
269 const bool is_audio = tok == Buffers::ProcessingToken::AUDIO_BACKEND
270 || tok == Buffers::ProcessingToken::AUDIO_PARALLEL;
271
272 InspectResult tok_result = is_audio
273 ? root_audio_buffer(tok, layer, context, window, cursor, x_min, x_max, row_h, 1)
274 : root_graphics_buffer(tok, layer, context, window, cursor, x_min, x_max, row_h, 1);
275
276 layer.relate(result.group.header.header_id, tok_result.group.header.header_id);
277 result.children.push_back(std::move(tok_result));
278 }
279
280 if (in_count > 0) {
281 const std::string in_label = "inputs [" + std::to_string(in_count) + "]";
282 auto in_group = make_value_group(
283 in_label, {},
284 layer, context, window, cursor,
285 x_min + k_inspect_indent, x_max, row_h, false);
286
287 InspectResult in_result;
288 in_result.group = std::move(in_group);
289 layer.relate(result.group.header.header_id, in_result.group.header.header_id);
290
291 for (uint32_t ch = 0; ch < in_count; ++ch) {
292 auto buf = m_bm.get_input_buffer(ch);
293 auto buf_result = buffer(
294 std::dynamic_pointer_cast<Buffers::Buffer>(buf), layer, context, window, cursor,
295 x_min, x_max, row_h, 2);
296 layer.relate(in_result.group.header.header_id, buf_result.group.header.header_id);
297 in_result.children.push_back(std::move(buf_result));
298 }
299
300 result.children.push_back(std::move(in_result));
301 }
302
303 return result;
304}
