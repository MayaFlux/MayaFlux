129glm::vec2 Context::to_ndc(double px, double py) const noexcept
130{
131 const auto& s = m_window->get_state();
132 return {
133 (static_cast<float>(px) / static_cast<float>(s.current_width)) * 2.0F - 1.0F,
134 1.0F - (static_cast<float>(py) / static_cast<float>(s.current_height)) * 2.0F
135 };

138void Context::handle_move(double px, double py)
139{
140 const glm::vec2 ndc = to_ndc(px, py);
141 const auto hit = m_layer->hit_test(ndc);
142
143 if (hit != m_hovered) {
144 if (m_hovered) {
145 auto it = m_callbacks.find(*m_hovered);
146 if (it != m_callbacks.end() && it->second.leave)
147 it->second.leave(*m_hovered);
148 }
149 if (hit) {
150 auto it = m_callbacks.find(*hit);
151 if (it != m_callbacks.end() && it->second.enter)
152 it->second.enter(*hit);
153 }
154 m_hovered = hit;
155 }
156
157 if (hit) {
158 auto it = m_callbacks.find(*hit);
159 if (it != m_callbacks.end() && it->second.move)
160 it->second.move(*hit, ndc);
161 }
162}

164void Context::handle_press(double px, double py, IO::MouseButtons btn)
165{
166 const glm::vec2 ndc = to_ndc(px, py);
167 const auto hit = m_layer->hit_test(ndc);
168 if (!hit)
169 return;
170
171 auto it = m_callbacks.find(*hit);
172 if (it == m_callbacks.end())
173 return;
174
175 auto btn_it = it->second.press.find(static_cast<int>(btn));
176 if (btn_it != it->second.press.end() && btn_it->second)
177 btn_it->second(*hit, ndc);
178}

196void Context::handle_scroll(double dx, double dy)
197{
198 if (!m_hovered)
199 return;
200
201 auto it = m_callbacks.find(*m_hovered);
202 if (it == m_callbacks.end() || !it->second.scroll)
203 return;
204
205 const auto [px, py] = m_window->get_event_source().get_mouse_position();
206 it->second.scroll(*m_hovered, to_ndc(px, py), dx, dy);
207}
