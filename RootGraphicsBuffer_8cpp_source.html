21void GraphicsBatchProcessor::processing_function(const std::shared_ptr<Buffer>& buffer)
22{
23 auto root_buf = std::dynamic_pointer_cast<RootGraphicsBuffer>(buffer);
24 if (!root_buf || root_buf != m_root_buffer) {
25 MF_RT_ERROR(Journal::Component::Core, Journal::Context::BufferProcessing,
26 "GraphicsBatchProcessor can only process its associated RootGraphicsBuffer");
27 return;
28 }
29
30 root_buf->cleanup_marked_buffers();
31
32 for (auto& ch_buffer : root_buf->get_child_buffers()) {
33 if (!ch_buffer)
34 continue;
35
36 if (ch_buffer->needs_removal()) {
37 continue;
38 }
39
40 if (!ch_buffer->has_data_for_cycle()) {
41 continue;
42 }
43
44 try {
45 if (ch_buffer->needs_default_processing() && ch_buffer->get_default_processor()) {
46 ch_buffer->process_default();
47 }
48
49 if (auto chain = ch_buffer->get_processing_chain()) {
50 if (ch_buffer->has_data_for_cycle()) {
51 chain->process_complete(ch_buffer);
52 }
53 }
54
55 auto vk_buffer = std::dynamic_pointer_cast<Buffers::VKBuffer>(ch_buffer);
56 if (vk_buffer && vk_buffer->has_render_pipeline()) {
57 for (const auto& [id, window] : vk_buffer->get_render_pipelines()) {
58 RootGraphicsBuffer::RenderableBufferInfo info;
59 info.buffer = vk_buffer;
60 info.target_window = window;
61 info.pipeline_id = id;
62 info.command_buffer_id = vk_buffer->get_pipeline_command(id);
63
64 root_buf->add_renderable_buffer(info);
65
66 MF_RT_TRACE(Journal::Component::Core, Journal::Context::BufferProcessing,
67 "Registered buffer for rendering to window '{}'",
68 window->get_create_info().title);
69 }
70 }
71
72 } catch (const std::exception& e) {
73 MF_RT_ERROR(Journal::Component::Core, Journal::Context::BufferProcessing,
74 "Error processing graphics buffer: {}", e.what());
75 }
76 }

202void PresentProcessor::fallback_renderer(const std::shared_ptr<RootGraphicsBuffer>& root)
203{
204 const auto& renderable_buffers = root->get_renderable_buffers();
205 if (renderable_buffers.empty()) {
206 if (root->get_buffer_count() > 0) {
207 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
208 "No renderable buffers found, but root has {} child buffers. Falling back to no-op.",
209 root->get_buffer_count());
210 }
211 return;
212 }
213
214 std::unordered_map<Core::Window*, std::vector<const RootGraphicsBuffer::RenderableBufferInfo*>> buffers_by_window;
215 for (const auto& renderable : renderable_buffers) {
216 if (renderable.target_window && renderable.target_window->is_graphics_registered()
217 && renderable.command_buffer_id != Portal::Graphics::INVALID_COMMAND_BUFFER) {
218 buffers_by_window[renderable.target_window.get()].push_back(&renderable);
219 }
220 }
221
222 if (buffers_by_window.empty()) {
223 root->clear_renderable_buffers();
224 return;
225 }
226
227 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
228 "PresentProcessor submitting to {} windows", buffers_by_window.size());
229
230 auto& foundry = Portal::Graphics::get_shader_foundry();
231 auto& flow = Portal::Graphics::get_render_flow();
232 auto display_service = Registry::BackendRegistry::instance()
233 .get_service<Registry::Service::DisplayService>();
234
235 if (!display_service) {
236 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
237 "DisplayService not available for dynamic rendering");
238 return;
239 }
240
241 for (const auto& [window_ptr, buffer_infos] : buffers_by_window) {
242 auto window = buffer_infos[0]->target_window;
243
244 uint64_t image_bits = display_service->acquire_next_swapchain_image(window);
245 if (image_bits == 0) {
246 MF_RT_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
247 "Failed to acquire swapchain image for window '{}'",
248 window->get_create_info().title);
249 continue;
250 }
251 vk::Image swapchain_image { reinterpret_cast<VkImage>(image_bits) };
252
253 auto primary_cmd_id = foundry.begin_commands(Portal::Graphics::ShaderFoundry::CommandBufferType::GRAPHICS);
254 auto primary_cmd = foundry.get_command_buffer(primary_cmd_id);
255
256 if (!primary_cmd) {
257 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
258 "Failed to create primary command buffer for window '{}'",
259 window->get_create_info().title);
260 for (const auto* info : buffer_infos) {
261 info->buffer->clear_pipeline_commands();
262 }
263 continue;
264 }
265
266 try {
267 flow.begin_rendering(primary_cmd_id, window, swapchain_image);
268
269 std::vector<vk::CommandBuffer> secondary_buffers;
270 for (const auto* info : buffer_infos) {
271 auto secondary = foundry.get_command_buffer(info->command_buffer_id);
272 if (secondary) {
273 secondary_buffers.push_back(secondary);
274 }
275 }
276
277 if (!secondary_buffers.empty()) {
278 primary_cmd.executeCommands(secondary_buffers);
279 }
280
281 flow.end_rendering(primary_cmd_id, window);
282
283 foundry.end_commands(primary_cmd_id);
284 uint64_t primary_bits = *reinterpret_cast<uint64_t*>(&primary_cmd);
285 display_service->submit_and_present(window, primary_bits);
286
287 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
288 "Presented {} buffers to window '{}'",
289 secondary_buffers.size(), window->get_create_info().title);
290
291 } catch (const std::exception& e) {
292 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
293 "Failed to submit/present for window '{}': {}",
294 window->get_create_info().title,
295 e.what());
296 }
297
298 for (const auto* info : buffer_infos) {
299 info->buffer->clear_pipeline_commands();
300 }
301 }
302
303 root->clear_renderable_buffers();
304}

337void RootGraphicsBuffer::cleanup_marked_buffers()
338{
339 if (m_pending_removal.empty()) {
340 return;
341 }
342
343 auto it = std::remove_if(
344 m_child_buffers.begin(),
345 m_child_buffers.end(),
346 [](const std::shared_ptr<Buffer>& buf) {
347 return buf && buf->needs_removal();
348 });
349
350 size_t removed_count = std::distance(it, m_child_buffers.end());
351
352 if (removed_count > 0) {
353 m_child_buffers.erase(it, m_child_buffers.end());
354
355 MF_INFO(Journal::Component::Core, Journal::Context::BufferManagement,
356 "Cleaned up {} graphics buffers (remaining: {})",
357 removed_count, m_child_buffers.size());
358 }
359
360 m_pending_removal.clear();
361}
