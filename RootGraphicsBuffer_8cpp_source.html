18void GraphicsBatchProcessor::processing_function(std::shared_ptr<Buffer> buffer)
19{
20 auto root_buf = std::dynamic_pointer_cast<RootGraphicsBuffer>(buffer);
21 if (!root_buf || root_buf != m_root_buffer) {
22 MF_RT_ERROR(Journal::Component::Core, Journal::Context::BufferProcessing,
23 "GraphicsBatchProcessor can only process its associated RootGraphicsBuffer");
24 return;
25 }
26
27 root_buf->cleanup_marked_buffers();
28
29 for (auto& ch_buffer : root_buf->get_child_buffers()) {
30 if (!ch_buffer)
31 continue;
32
33 if (ch_buffer->needs_removal()) {
34 continue;
35 }
36
37 if (!ch_buffer->has_data_for_cycle()) {
38 continue;
39 }
40
41 try {
42 if (ch_buffer->needs_default_processing() && ch_buffer->get_default_processor()) {
43 ch_buffer->process_default();
44 }
45
46 if (auto chain = ch_buffer->get_processing_chain()) {
47 if (ch_buffer->has_data_for_cycle()) {
48 chain->process(ch_buffer);
49 chain->process_final(ch_buffer);
50 }
51 }
52
53 auto vk_buffer = std::dynamic_pointer_cast<Buffers::VKBuffer>(ch_buffer);
54 if (vk_buffer && vk_buffer->has_render_pipeline()) {
55 for (const auto& [id, window] : vk_buffer->get_render_pipelines()) {
56 RootGraphicsBuffer::RenderableBufferInfo info;
57 info.buffer = vk_buffer;
58 info.target_window = window;
59 info.pipeline_id = id;
60 info.command_buffer_id = vk_buffer->get_pipeline_command(id);
61
62 root_buf->add_renderable_buffer(info);
63
64 MF_RT_TRACE(Journal::Component::Core, Journal::Context::BufferProcessing,
65 "Registered buffer for rendering to window '{}'",
66 window->get_create_info().title);
67 }
68 }
69
70 } catch (const std::exception& e) {
71 MF_RT_ERROR(Journal::Component::Core, Journal::Context::BufferProcessing,
72 "Error processing graphics buffer: {}", e.what());
73 }
74 }

200void PresentProcessor::fallback_renderer(const std::shared_ptr<RootGraphicsBuffer>& root)
201{
202 auto& foundry = Portal::Graphics::get_shader_foundry();
203 auto& flow = Portal::Graphics::get_render_flow();
204
205 const auto& renderable_buffers = root->get_renderable_buffers();
206
207 if (renderable_buffers.empty()) {
208 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
209 "No renderable buffers found in fallback renderer");
210 return;
211 }
212
213 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
214 "PresentProcessor submitting {} renderable buffers",
215 renderable_buffers.size());
216
217 std::unordered_map<Core::Window*, std::vector<const RootGraphicsBuffer::RenderableBufferInfo*>> buffers_by_window;
218
219 for (const auto& renderable : renderable_buffers) {
220 buffers_by_window[renderable.target_window.get()].push_back(&renderable);
221 }
222
223 for (const auto& [window_ptr, buffer_infos] : buffers_by_window) {
224 auto window = buffer_infos[0]->target_window;
225
226 if (!window) {
227 MF_RT_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
228 "Skipping present: window no longer valid");
229 continue;
230 }
231
232 try {
233 for (const auto* info : buffer_infos) {
234 if (info->command_buffer_id != Portal::Graphics::INVALID_COMMAND_BUFFER) {
235 flow.present_rendered_image(
236 info->command_buffer_id,
237 window);
238 }
239 }
240
241 MF_RT_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
242 "Presented {} buffers to window '{}'",
243 buffer_infos.size(),
244 window->get_create_info().title);
245
246 } catch (const std::exception& e) {
247 MF_RT_ERROR(Journal::Component::Buffers, Journal::Context::BufferProcessing,
248 "Failed to submit/present for window '{}': {}",
249 window->get_create_info().title,
250 e.what());
251 }
252
253 for (const auto* info : buffer_infos) {
254 info->buffer->clear_pipeline_commands();
255 }
256 }
257
258 root->clear_renderable_buffers();
259}

292void RootGraphicsBuffer::cleanup_marked_buffers()
293{
294 if (m_pending_removal.empty()) {
295 return;
296 }
297
298 auto it = std::remove_if(
299 m_child_buffers.begin(),
300 m_child_buffers.end(),
301 [](const std::shared_ptr<Buffer>& buf) {
302 return buf && buf->needs_removal();
303 });
304
305 size_t removed_count = std::distance(it, m_child_buffers.end());
306
307 if (removed_count > 0) {
308 m_child_buffers.erase(it, m_child_buffers.end());
309
310 MF_INFO(Journal::Component::Core, Journal::Context::BufferManagement,
311 "Cleaned up {} graphics buffers (remaining: {})",
312 removed_count, m_child_buffers.size());
313 }
314
315 m_pending_removal.clear();
316}
