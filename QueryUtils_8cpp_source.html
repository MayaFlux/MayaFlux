19 void write_row_quad(
20 std::vector<uint8_t>& bytes,
21 float x_min, float x_max, float bot, float top,
22 glm::vec3 bg)
23 {
24 const uint32_t stride = Kakshya::VertexLayout::for_meshes().stride_bytes;
25
26 const glm::vec3 bl { x_min, bot, 0.F };
27 const glm::vec3 br { x_max, bot, 0.F };
28 const glm::vec3 tl { x_min, top, 0.F };
29 const glm::vec3 tr { x_max, top, 0.F };
30 constexpr glm::vec3 white { 1.F, 1.F, 1.F };
31
32 auto write = [&](size_t i, glm::vec3 p, glm::vec3 c, float w, glm::vec2 uv) {
33 auto* v = bytes.data() + i * stride;
34 std::memcpy(v, &p, 12);
35 std::memcpy(v + 12, &c, 12);
36 std::memcpy(v + 24, &w, 4);
37 std::memcpy(v + 28, &uv, 8);
38 };
39
40 write(0, bl, bg, 0.F, {});
41 write(1, br, bg, 0.F, {});
42 write(2, tl, bg, 0.F, {});
43 write(3, br, bg, 0.F, {});
44 write(4, tr, bg, 0.F, {});
45 write(5, tl, bg, 0.F, {});
46
47 write(6, bl, white, 1.F, { 0.F, 1.F });
48 write(7, br, white, 1.F, { 1.F, 1.F });
49 write(8, tl, white, 1.F, { 0.F, 0.F });
50 write(9, br, white, 1.F, { 1.F, 1.F });
51 write(10, tr, white, 1.F, { 1.F, 0.F });
52 write(11, tl, white, 1.F, { 0.F, 0.F });
53 }

57std::pair<uint32_t, uint32_t> row_pixel_dims(
58 const std::shared_ptr<Core::Window>& window,
59 float x_min, float x_max, float row_h)
60{
61 const auto& ws = window->get_state();
62 const auto w = static_cast<uint32_t>(
63 (x_max - x_min) * 0.5F * static_cast<float>(ws.current_width));
64 const auto h = static_cast<uint32_t>(
65 row_h * 0.5F * static_cast<float>(ws.current_height));
66
67 const uint32_t min_h = Portal::Text::get_default_atlas().line_height();
68
69 return { std::max(w, 1U), std::max(h, min_h) };
70}

72ValueRow make_value_row(
73 const ValueSpec& spec,
74 Layer& layer,
75 const std::shared_ptr<Core::Window>& window,
76 LayoutCursor& cursor,
77 float x_min, float x_max, float row_h,
78 glm::vec3 bg)
79{
80 const auto [tw, th] = row_pixel_dims(window, x_min, x_max, row_h);
81
82 const std::string initial = spec.label + ": "
83 + (spec.reader ? spec.reader() : std::string {});
84
85 Portal::Text::PressParams params {
86 .color = { 1.F, 1.F, 1.F, 1.F },
87 .budget_h = th,
88 };
89
90 auto text_image = Portal::Text::press(initial, { tw, th }, params);
91 auto staging = Buffers::create_image_staging_buffer(text_image->get_size_bytes());
92
93 const uint32_t stride = Kakshya::VertexLayout::for_meshes().stride_bytes;
94 auto buf = Portal::Forma::create_buffer(
95 window, static_cast<size_t>(12) * stride,
96 Portal::Graphics::PrimitiveTopology::TRIANGLE_LIST,
97 { { "text", text_image } });
98
99 const float top = cursor.y();
100 const float bot = top - row_h;
101 cursor.advance(row_h);
102
103 std::vector<uint8_t> bytes(static_cast<size_t>(12) * stride, 0);
104 write_row_quad(bytes, x_min, x_max, bot, top, bg);
105 buf->submit(bytes);
106
107 Element el;
108 el.buffer = buf;
109 el.bounds_hint = Kinesis::AABB2D { .min = { x_min, bot }, .max = { x_max, top } };
110 el.interactive = false;
111 el.name = spec.label;
112 const uint32_t id = layer.add(std::move(el));
113
114 Link link(
115 [] { },
116 [reader = spec.reader,
117 label = spec.label,
118 text_image,
119 buf,
120 staging,
121 params]() mutable {
122 if (!buf || !reader)
123 return;
124 Portal::Text::repress(text_image,
125 label + ": " + reader(),
126 params, staging);
127 buf->bind_texture(0, text_image);
128 });
129
130 return ValueRow {
131 .element_id = id,
132 .buf = buf,
133 .text = text_image,
134 .link = std::move(link),
135 };
136}

138ValueGroup make_value_group(
139 std::string_view header_label,
140 std::span<const ValueSpec> values,
141 Layer& layer,
142 Context& context,
143 const std::shared_ptr<Core::Window>& window,
144 LayoutCursor& cursor,
145 float x_min, float x_max, float row_h,
146 bool initially_open)
147{
148 const auto [tw, th] = row_pixel_dims(window, x_min, x_max, row_h);
149
150 auto header_text = Portal::Text::press(
151 header_label,
152 { tw, th },
153 Portal::Text::PressParams {
154 .color = { 1.F, 1.F, 1.F, 1.F },
155 .budget_h = th,
156 });
157
158 auto header = make_collapsible(
159 window, layer, context, get_bridge(), cursor,
160 x_min, x_max, row_h, initially_open,
161 glm::vec3(0.25F), glm::vec3(0.35F),
162 header_text);
163
164 std::vector<ValueRow> rows;
165 rows.reserve(values.size());
166 for (const auto& spec : values) {
167 auto row = make_value_row(spec, layer, window, cursor,
168 x_min, x_max, row_h);
169 header.attach(layer, row.element_id);
170 rows.push_back(std::move(row));
171 }
172
173 header.cursor_out = cursor;
174
175 return ValueGroup {
176 .header = std::move(header),
177 .rows = std::move(rows),
178 };
179}
