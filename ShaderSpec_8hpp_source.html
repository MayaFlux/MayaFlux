30/**
31 * @struct KernelSource
32 * @brief Parsed representation of a user-supplied kernel lambda.
33 *
34 * Produced by the MF_KERNEL macro, which stringifies the lambda text
35 * at the call site. The parameter names must match binding and push
36 * constant field names declared on the enclosing ShaderSpec in
37 * declaration order: SSBO bindings first, then PC fields, then
38 * optionally a uint32_t index parameter named i.
39 *
40 * The body is extracted verbatim between the outermost braces and
41 * injected into the generated GLSL compute shader by the emitter.
42 *
43 * @code
44 * auto spec = ShaderSpec::Assemble{}
45 * .ssbo("sig", BindingDirection::InOut, Kakshya::GpuDataFormat::FLOAT32)
46 * .pc("gain")
47 * .kernel(MF_KERNEL([](float* sig, float gain, uint32_t i) {
48 * sig[i] = sig[i] * gain + sin(float(i) * 0.01f);
49 * }))
50 * .build();
51 * @endcode
52 */
53struct KernelSource {
54 std::string raw;
55 std::vector<std::string> param_names;
56 std::string body;
57
58 /**
59 * @brief Parse parameter names and body from a stringified lambda.
60 *
61 * Extracts identifiers from the parameter list by stripping type
62 * tokens (everything up to and including the last space or * before
63 * the identifier). Extracts body as the text between the first { and
64 * the matching closing }.
65 *
66 * @param stringified Raw text produced by MF_KERNEL.
67 * @return Populated KernelSource.
68 */
69 [[nodiscard]] static KernelSource parse(std::string_view stringified)
70 {
71 KernelSource ks;
72 ks.raw = std::string(stringified);
73
74 ///< Locate parameter list: between first '(' after '[' and its matching ')'
75 const auto bracket = stringified.find('[');
76 const auto paren_open = stringified.find('(', bracket);
77 if (paren_open == std::string_view::npos)
78 return ks;
79
80 std::size_t depth = 1;
81 std::size_t paren_close = paren_open + 1;
82 while (paren_close < stringified.size() && depth > 0) {
83 if (stringified[paren_close] == '(') {
84 ++depth;
85 } else if (stringified[paren_close] == ')') {
86 --depth;
87 }
88 ++paren_close;
89 }
90 --paren_close; ///< points at the closing ')'
91
92 const auto params_text = stringified.substr(paren_open + 1, paren_close - paren_open - 1);
93
94 ///< Split on ',' and extract the last token (the identifier) from each param
95 auto extract_name = [](std::string_view param) -> std::string {
96 ///< strip trailing whitespace
97 auto end = param.find_last_not_of(" \t\n\r");
98 if (end == std::string_view::npos)
99 return {};
100 param = param.substr(0, end + 1);
101 ///< find last space or * - identifier follows
102 auto sep = param.find_last_of(" *\t");
103 if (sep == std::string_view::npos)
104 return std::string(param);
105 return std::string(param.substr(sep + 1));
106 };
107
108 std::size_t start = 0;
109 while (start < params_text.size()) {
110 ///< find next comma not inside <> or ()
111 int angle = 0, paren = 0;
112 std::size_t pos = start;
113 while (pos < params_text.size()) {
114 char c = params_text[pos];
115 if (c == '<') {
116 ++angle;
117 } else if (c == '>') {
118 --angle;
119 } else if (c == '(') {
120 ++paren;
121 } else if (c == ')') {
122 --paren;
123 } else if (c == ',' && angle == 0 && paren == 0) {
124 break;
125 }
126 ++pos;
127 }
128 auto name = extract_name(params_text.substr(start, pos - start));
129 if (!name.empty())
130 ks.param_names.push_back(std::move(name));
131 start = (pos < params_text.size()) ? pos + 1 : pos;
132 }
133
134 ///< Locate body: between first '{' after ')' and its matching '}'
135 const auto brace_open = stringified.find('{', paren_close);
136 if (brace_open == std::string_view::npos)
137 return ks;
138
139 depth = 1;
140 std::size_t brace_close = brace_open + 1;
141 while (brace_close < stringified.size() && depth > 0) {
142 if (stringified[brace_close] == '{') {
143 ++depth;
144 } else if (stringified[brace_close] == '}') {
145 --depth;
146 }
147 ++brace_close;
148 }
149 --brace_close;
150
151 ks.body = std::string(stringified.substr(brace_open + 1, brace_close - brace_open - 1));
152 return ks;
153 }
154};

156/**
157 * @enum KernelOp
158 * @brief Named operation the SPIR-V emitter knows how to lower to opcodes.
159 *
160 * Each value maps to a fixed, unambiguous instruction sequence. No string
161 * parsing or expression compilation is involved.
162 *
163 * Naming convention: operands are listed in application order.
164 * PC fields are consumed in declaration order from ShaderSpec::pc_fields.
165 * SSBO operands are the InOut/Output bindings in declaration order.
166 *
167 * Single-SSBO elementwise — arithmetic (sig[i] = f(sig[i], pc...)):
168 * Scale sig[i] = sig[i] * pc[0]
169 * ScaleOffset sig[i] = sig[i] * pc[0] + pc[1]
170 * Fma sig[i] = sig[i] * pc[0] + pc[1] (fused, single instruction)
171 * Offset sig[i] = sig[i] + pc[0]
172 * Clip sig[i] = clamp(sig[i], pc[0], pc[1])
173 * Abs sig[i] = abs(sig[i])
174 * Negate sig[i] = -sig[i]
175 * Floor sig[i] = floor(sig[i])
176 * Ceil sig[i] = ceil(sig[i])
177 * Round sig[i] = round(sig[i])
178 * Trunc sig[i] = trunc(sig[i])
179 * Fract sig[i] = fract(sig[i])
180 * Sqrt sig[i] = sqrt(sig[i])
181 * InverseSqrt sig[i] = 1/sqrt(sig[i])
182 *
183 * Single-SSBO elementwise — transcendental (sig[i] = f(sig[i])):
184 * Sin sig[i] = sin(sig[i])
185 * Cos sig[i] = cos(sig[i])
186 * Tan sig[i] = tan(sig[i])
187 * Asin sig[i] = asin(sig[i])
188 * Acos sig[i] = acos(sig[i])
189 * Atan sig[i] = atan(sig[i])
190 * Sinh sig[i] = sinh(sig[i])
191 * Cosh sig[i] = cosh(sig[i])
192 * Tanh sig[i] = tanh(sig[i])
193 * Exp sig[i] = e^sig[i]
194 * Exp2 sig[i] = 2^sig[i]
195 * Log sig[i] = ln(sig[i])
196 * Log2 sig[i] = log2(sig[i])
197 *
198 * Two-SSBO elementwise (out[i] = f(a[i], b[i])):
199 * Add out[i] = a[i] + b[i]
200 * Multiply out[i] = a[i] * b[i]
201 * Mix out[i] = a[i] + (b[i] - a[i]) * pc[0]
202 * Sub out[i] = a[i] - b[i]
203 * Pow out[i] = pow(a[i], b[i])
204 * Atan2 out[i] = atan(a[i], b[i])
205 * Min out[i] = min(a[i], b[i])
206 * MaxTwo out[i] = max(a[i], b[i])
207 * Step out[i] = step(a[i], b[i]) edge=a, x=b
208 *
209 * Two-SSBO + one PC:
210 * SmoothStep out[i] = smoothstep(a[i], b[i], pc[0])
211 *
212 * Reduction operations (one InOut SSBO, shared memory):
213 * Sum accumulate + into shared[lid], tree reduce
214 * Max accumulate max into shared[lid], tree reduce
215 */
216enum class KernelOp : uint8_t {
217 // arithmetic
218 Scale,
219 ScaleOffset,
220 Fma,
221 Offset,
222 Clip,
223 Abs,
224 Negate,
225 Floor,
226 Ceil,
227 Round,
228 Trunc,
229 Fract,
230 Sqrt,
231 InverseSqrt,
232 // transcendental
233 Sin,
234 Cos,
235 Tan,
236 Asin,
237 Acos,
238 Atan,
239 Sinh,
240 Cosh,
241 Tanh,
242 Exp,
243 Exp2,
244 Log,
245 Log2,
246 // two-SSBO
247 Add,
248 Multiply,
249 Mix,
250 Sub,
251 Pow,
252 Atan2,
253 Min,
254 MaxTwo,
255 Step,
256 SmoothStep,
257 // reduction
258 Sum,
