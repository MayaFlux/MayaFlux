26/**
27 * @file Forma.hpp
28 * @brief Factory free functions for the Forma surface system.
29 *
30 * Call initialize() once after engine startup to store BufferManager,
31 * TaskScheduler, and EventManager references. Subsequent factory calls
32 * require only the arguments that legitimately vary per call (Window,
33 * geometry function, initial value). This matches the Portal::Text and
34 * Portal::Graphics initialization contracts.
35 *
36 * Typical sequence:
37 * @code
38 * Portal::Forma::initialize(buffer_manager, scheduler, event_manager);
39 *
40 * auto [layer, ctx] = Forma::create_layer(window, "hud");
41 *
42 * auto el = Forma::create_element<float>(
43 * *layer, window, geom_fn, 0.5f);
44 *
45 * auto bridge = Forma::create_bridge();
46 * bridge.bind(el.state, envelope);
47 * bridge.write(el.state, compute_proc, offsetof(PC, cutoff));

141 * BufferManager is taken from the stored initialize() state.
142 * Returns the fully constructed Mapped<T>. The caller holds it.
143 * element.id is stable and can be passed to Context callbacks and Bridge.
144 *
145 * @tparam T MappedState value type: float, glm::vec2, etc.
146 * @param layer Layer to register the element on.
147 * @param window Target window for rendering.
148 * @param geom Geometry function producing vertex bytes from T.
149 * @param initial Starting value written into MappedState.
150 * @param capacity Initial FormaBuffer capacity in bytes.
151 * @param topology Primitive topology for the FormaBuffer.
152 * @return Fully constructed Mapped<T> with element registered in @p layer.
153 */
154template <typename T>
155[[nodiscard]] Mapped<T> create_element(
156 Layer& layer,
157 std::shared_ptr<Core::Window> window,
158 GeometryFn<T> geom,
159 T initial,
160 size_t capacity = 4096,
161 Graphics::PrimitiveTopology topology = Graphics::PrimitiveTopology::TRIANGLE_STRIP)
162{
163 auto buf = create_buffer(std::move(window), capacity, topology);
164 auto mapped = make_mapped<T>(initial, std::move(geom), buf);
165 mapped.element.id = layer.add(mapped.element);
166 get_bridge().register_element(mapped);
167 return mapped;
168}
