104 void apply_if_ready()
105 {
106 if constexpr (std::is_base_of_v<Kakshya::SignalSourceContainer, T>) {
107 if (m_ctx.domain) {
108 apply_container_context();
109 m_ctx = CreationContext {};
110 }
111 } else if (m_ctx.domain && (m_ctx.channel || m_ctx.channels)) {
112 if constexpr (std::is_base_of_v<Nodes::Node, T>) {
113 if (m_ctx.domain) {
114 apply_node_context();
115 }
116 } else if constexpr (std::is_base_of_v<Buffers::Buffer, T>) {
117 if (m_ctx.domain) {
118 apply_buffer_context();
119 }
120 } else if constexpr (std::is_base_of_v<Nodes::Network::NodeNetwork, T>) {
121 if (m_ctx.domain) {
122 apply_network_context();
123 }
124 }
125 m_ctx = CreationContext {};
126 } else if (m_ctx.domain) {
127 if constexpr (std::is_base_of_v<Buffers::VKBuffer, T>) {
128 apply_buffer_context();
129 }
130 if (m_ctx.domain == Domain::GRAPHICS) {
131 if constexpr (std::is_base_of_v<Nodes::Node, T>) {
132 apply_node_context();
133 }
134 if constexpr (std::is_base_of_v<Nodes::Network::NodeNetwork, T>) {
135 apply_network_context();
136 }
137 }
138 m_ctx = CreationContext {};
139 }
140 }
