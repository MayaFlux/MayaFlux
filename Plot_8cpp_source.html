19std::vector<std::span<const double>> series_by_role(
20 const Kakshya::PlotContainer& container,
21 Role role)
22{
23 const auto& dims = container.get_structure().dimensions;
24 const auto& data = container.get_processed_data();
25
26 std::vector<std::span<const double>> result;
27 for (size_t i = 0; i < dims.size() && i < data.size(); ++i) {
28 if (dims[i].role != role)
29 continue;
30 const auto* vec = std::get_if<std::vector<double>>(&data[i]);
31 if (!vec || vec->empty())
32 continue;
33 result.emplace_back(vec->data(), vec->size());
34 }
35 return result;
36}

41
42std::pair<float, float> data_range(std::span<const double> series)
43{
44 if (series.empty())
45 return { 0.F, 1.F };
46
47 double lo = std::numeric_limits<double>::max();
48 double hi = std::numeric_limits<double>::lowest();
49 for (double v : series) {
50 if (v < lo)
51 lo = v;
52 if (v > hi)
53 hi = v;
54 }
55 if (lo == hi)
56 hi = lo + 1.0;
57 return { static_cast<float>(lo), static_cast<float>(hi) };
58}

60void apply_auto_scale(AxisRange& range,
61 const std::vector<std::span<const double>>& series)
62{
63 if (!range.auto_scale || series.empty())
64 return;
65
66 float lo = std::numeric_limits<float>::max();
67 float hi = std::numeric_limits<float>::lowest();
68 for (const auto& s : series) {
69 auto [slo, shi] = data_range(s);
70 lo = std::min(lo, slo);
71 hi = std::max(hi, shi);
72 }
73 range.min = lo;
74 range.max = hi;
75}

102 std::vector<float> uniform_x(size_t n, const AxisRange& x_range)
103 {
104 std::vector<float> xs(n);
105 if (n == 1) {
106 xs[0] = x_range.to_ndc((x_range.min + x_range.max) * 0.5F);
107 return xs;
108 }
109 for (size_t i = 0; i < n; ++i) {
110 const float t = static_cast<float>(i) / static_cast<float>(n - 1);
111 const float v = x_range.min + t * (x_range.max - x_range.min);
112 xs[i] = x_range.to_ndc(v);
113 }
114 return xs;
115 }

120// waveform
121// =============================================================================
122
123GeometryFn<std::shared_ptr<Kakshya::PlotContainer>> waveform(
124 AxisRange x_range,
125 AxisRange y_range,
126 AxisRange z_range,
127 const std::vector<glm::vec3>& palette,
128 float thickness)
129{
130 return [x_range, y_range, z_range, palette, thickness](
131 const std::shared_ptr<Kakshya::PlotContainer>& container,
132 std::vector<uint8_t>& out,
133 Element&) mutable {
134 if (!container)
135 return;
136
137 container->process_default();
138
139 auto y_series = series_by_role(*container, Role::SPATIAL_Y);
140 if (y_series.empty())
141 y_series = series_by_role(*container, Role::TIME);
142 if (y_series.empty())
143 return;
144
145 const auto x_series = series_by_role(*container, Role::SPATIAL_X);
146 const auto z_series = series_by_role(*container, Role::SPATIAL_Z);
147
148 apply_auto_scale(x_range, x_series.empty() ? y_series : x_series);
149 apply_auto_scale(y_range, y_series);
150 apply_auto_scale(z_range, z_series);
151
152 out.clear();
153
154 for (size_t si = 0; si < y_series.size(); ++si) {
155 const auto& ys = y_series[si];
156 const size_t n = ys.size();
157 if (n == 0)
158 continue;
159
160 const glm::vec3 color = palette_color(palette, si);
161
162 const bool has_x = si < x_series.size() && x_series[si].size() == n;
163 const bool has_z = si < z_series.size() && z_series[si].size() == n;
164
165 std::vector<float> xs;
166 if (has_x) {
167 xs.reserve(n);
168 for (double v : x_series[si])
169 xs.push_back(x_range.to_ndc(static_cast<float>(v)));
170 } else {
171 xs = uniform_x(n, x_range);
172 }
173
174 for (size_t i = 0; i < n; ++i) {
175 LineVertex v;
176 v.position = glm::vec3(
177 xs[i],
178 y_range.to_ndc(static_cast<float>(ys[i])),
179 has_z ? z_range.to_ndc(static_cast<float>(z_series[si][i])) : 0.F);
180 v.color = color;
181 v.thickness = thickness;
182 push_vertex(out, v);
183 }
184
185 // Degenerate separator between runs: repeat last vertex twice
186 // with zero thickness so the LINE_STRIP draws no segment between runs.
187 if (si + 1 < y_series.size()) {
188 const float last_z = has_z
189 ? z_range.to_ndc(static_cast<float>(z_series[si][n - 1]))
190 : 0.F;
191 LineVertex sep;
192 sep.position = glm::vec3(
193 xs[n - 1],
194 y_range.to_ndc(static_cast<float>(ys[n - 1])),
195 last_z);
196 sep.color = color;
197 sep.thickness = 0.F;
198 push_vertex(out, sep);
199 push_vertex(out, sep);
200 }
201 }
202 };
203}

209GeometryFn<std::shared_ptr<Kakshya::PlotContainer>> scatter(
210 AxisRange x_range,
211 AxisRange y_range,
212 AxisRange z_range,
213 const std::vector<glm::vec3>& palette,
214 float point_size)
215{
216 return [x_range, y_range, z_range, palette, point_size](
217 const std::shared_ptr<Kakshya::PlotContainer>& container,
218 std::vector<uint8_t>& out,
219 Element&) mutable {
220 if (!container)
221 return;
222
223 container->process_default();
224
225 const auto x_series = series_by_role(*container, Role::SPATIAL_X);
226 auto y_series = series_by_role(*container, Role::SPATIAL_Y);
227 const auto z_series = series_by_role(*container, Role::SPATIAL_Z);
228 const auto c_series = series_by_role(*container, Role::COLOR);
229
230 if (y_series.empty())
231 return;
232
233 apply_auto_scale(x_range, x_series);
234 apply_auto_scale(y_range, y_series);
235 apply_auto_scale(z_range, z_series);
236
237 out.clear();
238
239 for (size_t si = 0; si < y_series.size(); ++si) {
240 const auto& ys = y_series[si];
241 const size_t n = ys.size();
242 if (n == 0)
243 continue;
244
245 const glm::vec3 base_color = palette_color(palette, si);
246
247 // Reuse last X series if fewer X series than Y series.
248 const bool has_x = !x_series.empty();
249 const auto& xs = has_x
250 ? x_series[std::min(si, x_series.size() - 1)]
251 : std::span<const double> {};
252
253 const bool has_z = si < z_series.size() && z_series[si].size() == n;
254 const bool has_c = si < c_series.size() && c_series[si].size() == n;
255
256 for (size_t i = 0; i < n; ++i) {
257 PointVertex v;
258
259 v.position.x = has_x
260 ? x_range.to_ndc(static_cast<float>(xs[std::min(i, xs.size() - 1)]))
261 : x_range.to_ndc(x_range.min
262 + static_cast<float>(i)
263 / static_cast<float>(std::max<size_t>(n - 1, 1))
264 * (x_range.max - x_range.min));
265
266 v.position.y = y_range.to_ndc(static_cast<float>(ys[i]));
267
268 v.position.z = has_z
269 ? z_range.to_ndc(static_cast<float>(z_series[si][i]))
270 : 0.F;
271
272 if (has_c) {
273 const float t = y_range.normalise(
274 static_cast<float>(c_series[si][i]));
275 v.color = glm::mix(glm::vec3(1.F), base_color, t);
276 } else {
277 v.color = base_color;
278 }
279
280 v.size = point_size;
281 push_vertex(out, v);
282 }
283 }
284 };

288// bars
289// =============================================================================
290
291GeometryFn<std::shared_ptr<Kakshya::PlotContainer>> bars(
292 AxisRange x_range,
293 AxisRange y_range,
294 const std::vector<glm::vec3>& palette)
295{
296 return [x_range, y_range, palette](
297 const std::shared_ptr<Kakshya::PlotContainer>& container,
298 std::vector<uint8_t>& out,
299 Element&) mutable {
300 if (!container)
301 return;
302
303 container->process_default();
304
305 auto f_series = series_by_role(*container, Role::FREQUENCY);
306 if (f_series.empty())
307 f_series = series_by_role(*container, Role::SPATIAL_Y);
308 if (f_series.empty())
309 return;
310
311 apply_auto_scale(y_range, f_series);
312
313 out.clear();
314
315 for (size_t si = 0; si < f_series.size(); ++si) {
316 const auto& series = f_series[si];
317 const size_t n = series.size();
318 if (n == 0)
319 continue;
320
321 const glm::vec3 color = palette_color(palette, si);
322
323 const float bar_w = (x_range.max - x_range.min)
324 / static_cast<float>(n);
325 const float y_base = y_range.to_ndc(0.F);
326
327 for (size_t i = 0; i < n; ++i) {
328 const float x_left = x_range.to_ndc(
329 x_range.min + static_cast<float>(i) * bar_w);
330 const float x_right = x_range.to_ndc(
331 x_range.min + static_cast<float>(i + 1) * bar_w);
332 const float y_top = y_range.to_ndc(
333 static_cast<float>(series[i]));
334
335 const auto emit = [&](float x, float y) {
336 MeshVertex v;
337 v.position = glm::vec3(x, y, 0.F);
338 v.color = color;
339 push_vertex(out, v);
340 };
341
342 // Two triangles per bar (TRIANGLE_LIST).
343 emit(x_left, y_base);
344 emit(x_right, y_base);
345 emit(x_left, y_top);
346 emit(x_right, y_base);
347 emit(x_right, y_top);
348 emit(x_left, y_top);
349 }
350 }
351 };
352}

358GeometryFn<float> background(
359 Kinesis::AABB2D bounds,
360 glm::vec3 color,
361 const std::shared_ptr<Core::VKImage>& texture)
362{
363 return [bounds, color, texture](
364 float,
365 std::vector<uint8_t>& out,
366 Element& el) {
367 out.clear();
368
369 if (texture) {
370 // Textured quad — four TextureQuadVertex as TRIANGLE_STRIP.
371 // UV origin bottom-left matching Vulkan image layout.
372 const std::array<TextureQuadVertex, 4> verts = { {
373 { .position = { bounds.min.x, bounds.min.y, 0.F }, .texcoord = { 0.F, 1.F } },
374 { .position = { bounds.max.x, bounds.min.y, 0.F }, .texcoord = { 1.F, 1.F } },
375 { .position = { bounds.min.x, bounds.max.y, 0.F }, .texcoord = { 0.F, 0.F } },
376 { .position = { bounds.max.x, bounds.max.y, 0.F }, .texcoord = { 1.F, 0.F } },
377 } };
378 const auto* src = reinterpret_cast<const uint8_t*>(verts.data());
379 out.insert(out.end(), src, src + sizeof(verts));
380 } else {
381 // Solid color quad — four MeshVertex as TRIANGLE_STRIP.
382 const std::array<MeshVertex, 4> verts = { {
383 { .position = { bounds.min.x, bounds.min.y, 0.F }, .color = color },
384 { .position = { bounds.max.x, bounds.min.y, 0.F }, .color = color },
385 { .position = { bounds.min.x, bounds.max.y, 0.F }, .color = color },
386 { .position = { bounds.max.x, bounds.max.y, 0.F }, .color = color },
387 } };
388 const auto* src = reinterpret_cast<const uint8_t*>(verts.data());
389 out.insert(out.end(), src, src + sizeof(verts));
390 }
391
392 el.bounds_hint = bounds;
393 };
