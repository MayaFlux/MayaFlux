80double Phasor::process_sample(double input)
81{
82 double output = 0.0;
83
84 double effective_freq = m_frequency;
85
86 if (m_frequency_modulator) {
87 atomic_inc_modulator_count(m_frequency_modulator->m_modulator_count, 1);
88 uint32_t state = m_frequency_modulator->m_state.load();
89 if (state & NodeState::PROCESSED) {
90 effective_freq += m_frequency_modulator->get_last_output();
91 } else {
92 effective_freq = m_frequency_modulator->process_sample(0.F);
93 atomic_add_flag(m_frequency_modulator->m_state, NodeState::PROCESSED);
94 }
95
96 update_phase_increment(effective_freq);
97 }
98
99 output = m_phase * m_amplitude;
100
101 if (m_amplitude_modulator) {
102 atomic_inc_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
103 uint32_t state = m_amplitude_modulator->m_state.load();
104 if (state & NodeState::PROCESSED) {
105 output *= m_amplitude_modulator->get_last_output();
106 } else {
107 output *= m_amplitude_modulator->process_sample(0.F);
108 atomic_add_flag(m_amplitude_modulator->m_state, NodeState::PROCESSED);
109 }
110 }
111
112 output += m_offset;
113
114 m_phase += m_phase_inc;
115 if (m_phase >= 1.0) {
116 m_phase -= 1.0;
117 m_phase_wrapped = true;
118 }
119
120 m_last_output = output;
121
122 if ((!m_state_saved || (m_state_saved && m_fire_events_during_snapshot))
123 && !m_networked_node) {
124 notify_tick(output);
125 }
126
127 if (m_frequency_modulator) {
128 atomic_dec_modulator_count(m_frequency_modulator->m_modulator_count, 1);
129 try_reset_processed_state(m_frequency_modulator);
130 }
131 if (m_amplitude_modulator) {
132 atomic_dec_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
133 try_reset_processed_state(m_amplitude_modulator);
134 }
135
136 return output;
137}

171void Phasor::on_threshold(const NodeHook& callback, double threshold, bool /*rising*/)
172{
173 std::pair<NodeHook, double> entry = std::make_pair(callback, threshold);

201void Phasor::notify_tick(double value)
202{
203 update_context(value);
204
205 auto& ctx = get_last_context();
206 for (auto& callback : m_callbacks) {
207 callback(ctx);
208 }
209
210 if (m_phase_wrapped) {
211 for (auto& callback : m_phase_wrap_callbacks) {
212 callback(ctx);
213 }
214 }
215
216 for (auto& [callback, condition] : m_conditional_callbacks) {
217 if (condition(ctx)) {
218 callback(ctx);
219 }
220 }
221
222 for (auto& [callback, threshold] : m_threshold_callbacks) {
223 if (value >= threshold && !m_threshold_crossed) {
224 callback(ctx);
225 m_threshold_crossed = true;
226 } else if (value < threshold) {
227 m_threshold_crossed = false;
228 }
229 }
230}
