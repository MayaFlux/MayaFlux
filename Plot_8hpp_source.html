13/**
14 * @struct AxisRange
15 * @brief Scalar domain extent for one plot axis.
16 *
17 * Maps raw double values in [min, max] to the normalised render space
18 * expected by the geometry function. When auto_scale is true, min and max
19 * are recomputed from the series data on every process() call and the
20 * stored values are ignored until then.
21 */
22struct AxisRange {
23 float min { -1.F };
24 float max { 1.F };
25 bool auto_scale { false };
26
27 /** @brief Map a value into [0, 1] within this range. */
28 [[nodiscard]] float normalise(float v) const noexcept
29 {
30 if (max == min)
31 return 0.F;
32 return (v - min) / (max - min);
33 }
34
35 /** @brief Map a value into [-1, 1] NDC within this range. */
36 [[nodiscard]] float to_ndc(float v) const noexcept
37 {
38 return normalise(v) * 2.F - 1.F;
39 }
40};

97/**
98 * @brief LINE_STRIP waveform for TIME or SPATIAL_Y series.
99 *
100 * Reads all SPATIAL_X series as explicit X sample positions. If none are
101 * present, generates uniform X positions across [x_range.min, x_range.max]
102 * for each Y series independently.
103 *
104 * SPATIAL_Z series map to the Z position of each vertex if present,
105 * enabling 3D ribbon / parametric curve display. If a SPATIAL_Z series

109 * degenerate vertex so they share one FormaBuffer.

111 * Per-series color is resolved from @p palette cyclically.

113 * Calls container->process_default() before reading processed_data.
114 *
115 * @param x_range Domain for X. auto_scale recomputes from data each frame.
116 * @param y_range Domain for Y.
117 * @param z_range Domain for Z. Ignored if no SPATIAL_Z series present.
118 * @param palette Per-series colors, cycled. {1,1,1} if empty.
119 * @param thickness Line thickness in pixels.
120 */
