94 }
95
96 MF_DEBUG(Journal::Component::Core, Journal::Context::Init,
97 "Network IO thread exiting");
98 });
99#else
100 m_io_stop_requested.store(false);
101 m_io_thread = std::thread([this]() {
102 while (!m_io_stop_requested.load()) {
103 try {
104 m_io_context->run();
105 break;
106 } catch (const std::exception& e) {
107 MF_ERROR(Journal::Component::Core, Journal::Context::Networking,
108 "IO context exception: {}", e.what());
109 }
110 }
111
112 MF_DEBUG(Journal::Component::Core, Journal::Context::Init,

133void NetworkSubsystem::stop()
134{
135 if (!m_running.load()) {
136 return;
137 }
138
139 m_running.store(false);
140
141 {
142 std::shared_lock lock(m_backends_mutex);
143 for (auto& [transport, backend] : m_backends) {
144 backend->stop();
145 }
146 }
147
148 m_work_guard.reset();
149 m_io_context->stop();
150
151#if MAYAFLUX_USE_JTHREAD
152 if (m_io_thread.joinable()) {
153 m_io_thread.request_stop();
154 m_io_thread.join();
155 }
156#else
157 m_io_stop_requested.store(true);
158 if (m_io_thread.joinable()) {
159 m_io_thread.join();
160 }
161#endif
162
163 m_io_context->restart();
164
165 MF_INFO(Journal::Component::Core, Journal::Context::Shutdown,
166 "Network Subsystem stopped");

207bool NetworkSubsystem::add_backend(std::unique_ptr<INetworkBackend> backend)
208{
209 if (!backend) {
210 return false;
211 }
212
213 NetworkTransport transport = backend->get_transport();
214
215 std::unique_lock lock(m_backends_mutex);
216
217 if (m_backends.contains(transport)) {
218 MF_WARN(Journal::Component::Core, Journal::Context::NetworkSubsystem,
219 "Network backend {} already registered", backend->get_name());
220 return false;
221 }
222
223 if (!backend->initialize()) {
224 MF_ERROR(Journal::Component::Core, Journal::Context::NetworkSubsystem,
225 "Failed to initialize network backend: {}", backend->get_name());
226 return false;
227 }
228
229 backend->set_receive_callback(
230 [this](uint64_t id, const uint8_t* data, size_t size, std::string_view addr) {
231 on_backend_receive(id, data, size, addr);
232 });
233
234 backend->set_state_callback(
235 [this](const EndpointInfo& info, EndpointState prev, EndpointState curr) {
236 on_backend_state_change(info, prev, curr);
237 });
238
239 if (auto* tcp = dynamic_cast<TCPBackend*>(backend.get())) {
240 tcp->set_endpoint_id_allocator([this]() -> uint64_t {
241 return m_next_endpoint_id.fetch_add(1);
242 });
243 }
244
245 MF_INFO(Journal::Component::Core, Journal::Context::NetworkSubsystem,
246 "Added network backend: {}", backend->get_name());
247
248 m_backends[transport] = std::move(backend);
249 return true;
250}

274uint64_t NetworkSubsystem::open_endpoint(const EndpointInfo& info)
275{
276 auto* backend = get_backend(info.transport);
277 if (!backend) {
278 MF_ERROR(Journal::Component::Core, Journal::Context::NetworkSubsystem,
279 "No backend for transport {}", static_cast<int>(info.transport));
280 return 0;
281 }
282
283 EndpointInfo ep = info;
284 ep.id = m_next_endpoint_id.fetch_add(1);
285
286 uint64_t backend_id = backend->open_endpoint(ep);
287 if (backend_id == 0) {
288 return 0;
289 }
290
291 {
292 std::unique_lock lock(m_routing_mutex);
293 m_endpoint_routing[ep.id] = info.transport;
294 }
295
296 return ep.id;
297}

456INetworkBackend* NetworkSubsystem::resolve_backend(uint64_t endpoint_id) const
457{
458 NetworkTransport transport {};
459 {
460 std::shared_lock lock(m_routing_mutex);
461 auto it = m_endpoint_routing.find(endpoint_id);
