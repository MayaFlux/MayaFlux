73double Sine::process_sample(double input)
74{
75 if (m_frequency_modulator) {
76 atomic_inc_modulator_count(m_frequency_modulator->m_modulator_count, 1);
77 uint32_t state = m_frequency_modulator->m_state.load();
78 double current_freq = m_frequency;
79
80 if (state & NodeState::PROCESSED) {
81 current_freq += m_frequency_modulator->get_last_output();
82 } else {
83 current_freq += m_frequency_modulator->process_sample(0.F);
84 atomic_add_flag(m_frequency_modulator->m_state, NodeState::PROCESSED);
85 }
86 update_phase_increment(current_freq);
87 }
88
89 double current_sample = std::sin(m_phase + m_offset);
90 m_phase += m_phase_inc;
91
92 if (m_phase > 2 * M_PI) {
93 m_phase -= 2 * M_PI;
94 } else if (m_phase < -2 * M_PI) {
95 m_phase += 2 * M_PI;
96 }
97
98 double current_amplitude = m_amplitude;
99 if (m_amplitude_modulator) {
100 atomic_inc_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
101 uint32_t state = m_amplitude_modulator->m_state.load();
102
103 if (state & NodeState::PROCESSED) {
104 current_amplitude += m_amplitude_modulator->get_last_output();
105 if (!(state & NodeState::ACTIVE)) {
106 atomic_remove_flag(m_amplitude_modulator->m_state, NodeState::PROCESSED);
107 }
108 } else {
109 current_amplitude += m_amplitude_modulator->process_sample(0.F);
110 atomic_add_flag(m_amplitude_modulator->m_state, NodeState::PROCESSED);
111 }
112 }
113 current_sample *= current_amplitude;
114
115 if (input != 0.0) {
116 current_sample += input;
117 current_sample *= 0.5F;
118 }
119
120 m_last_output = current_sample;
121
122 if ((!m_state_saved || (m_state_saved && m_fire_events_during_snapshot))
123 && !m_networked_node) {
124 notify_tick(current_sample);
125 }
126
127 if (m_frequency_modulator) {
128 atomic_dec_modulator_count(m_frequency_modulator->m_modulator_count, 1);
129 try_reset_processed_state(m_frequency_modulator);
130 }
131 if (m_amplitude_modulator) {
132 atomic_dec_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
133 try_reset_processed_state(m_amplitude_modulator);
134 }
135 return current_sample;
136}
