56void VisionProcessor::process(const std::shared_ptr<SignalSourceContainer>& container)
57{
58 if (m_width == 0 || m_height == 0 || !container)

63 if (auto vc = std::dynamic_pointer_cast<VideoStreamContainer>(container)) {
