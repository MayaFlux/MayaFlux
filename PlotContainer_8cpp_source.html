50 static_cast<int>(role),

220std::vector<DataVariant> PlotContainer::get_region_data(const Region& region) const
221{
222 if (region.start_coordinates.empty() || region.end_coordinates.empty())
223 return {};
224
225 const uint64_t series_idx = region.start_coordinates[0];
226 if (series_idx >= m_data.size())
227 return {};
228
229 const auto* src = std::get_if<std::vector<double>>(&m_data[series_idx]);
230 if (!src || src->empty())
231 return {};
232
233 const uint64_t s = (region.start_coordinates.size() > 1) ? region.start_coordinates[1] : 0;
234 const uint64_t e = (region.end_coordinates.size() > 1)
235 ? std::min(region.end_coordinates[1], static_cast<uint64_t>(src->size() - 1))
236 : static_cast<uint64_t>(src->size() - 1);
237
238 if (s > e)
239 return {};
240
241 std::vector<double> slice(src->begin() + static_cast<ptrdiff_t>(s),
242 src->begin() + static_cast<ptrdiff_t>(e + 1));
243 return { DataVariant(std::move(slice)) };
244}

246void PlotContainer::set_region_data(const Region& region, const std::vector<DataVariant>& data)
247{
248 if (data.empty() || region.start_coordinates.empty())
249 return;
250
251 const uint64_t series_idx = region.start_coordinates[0];
252 if (series_idx >= m_data.size())
253 return;
254
255 auto* dst = std::get_if<std::vector<double>>(&m_data[series_idx]);
256 if (!dst)
257 return;
258
259 const auto* src = std::get_if<std::vector<double>>(&data[0]);
260 if (!src || src->empty())
261 return;
262
263 const uint64_t s = (region.start_coordinates.size() > 1) ? region.start_coordinates[1] : 0;
264 const uint64_t e = (region.end_coordinates.size() > 1)
265 ? std::min(region.end_coordinates[1], static_cast<uint64_t>(dst->size() - 1))
266 : static_cast<uint64_t>(dst->size() - 1);
267
268 if (s > e)
269 return;
270
271 const uint64_t n = std::min<uint64_t>(src->size(), e - s + 1);
272 std::copy_n(src->begin(), n, dst->begin() + static_cast<ptrdiff_t>(s));
273}
