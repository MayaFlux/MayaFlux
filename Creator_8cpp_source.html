20void register_node(const std::shared_ptr<Nodes::Node>& node, const CreationContext& ctx)
21{
22 auto token = get_node_token(ctx.domain.value());
23
24 if (ctx.channel.has_value()) {
25 register_node(node, token, ctx.channel.value());
26 } else if (ctx.channels.has_value()) {
27 for (uint32_t ch : ctx.channels.value()) {
28 register_node(node, token, ch);
29 }
30 } else if (node->get_channel_mask() != 0) {
31 for (uint32_t ch = 0; ch < 32; ++ch) {
32 if (node->get_channel_mask() & (1 << ch)) {
33 register_node(node, token, ch);
34 }
35 }
36 } else {
37 register_node(node, token, 0);
38 }
39}

41void register_network(const std::shared_ptr<Nodes::Network::NodeNetwork>& network, const CreationContext& ctx)
42{
43 auto token = get_node_token(ctx.domain.value());
44
45 if (token == Nodes::ProcessingToken::AUDIO_RATE) {
46 if (
47 network->get_output_mode() != Nodes::Network::OutputMode::AUDIO_SINK
48 && network->get_output_mode() != Nodes::Network::OutputMode::AUDIO_COMPUTE) {
49 MF_WARN(Journal::Component::API,
50 Journal::Context::Init,
51 "Registering audio network in AUDIO_RATE domain without AUDIO_SINK or AUDIO_COMPUTE mode. Forcing AUDIO_SINK mode.");
52 network->set_output_mode(Nodes::Network::OutputMode::AUDIO_SINK);
53 }
54 if (ctx.channel.has_value()) {
55 network->add_channel_usage(ctx.channel.value());
56 } else if (ctx.channels.has_value()) {
57 for (uint32_t ch : ctx.channels.value()) {
58 network->add_channel_usage(ch);
59 }
60 }
61 } else if (token == Nodes::ProcessingToken::VISUAL_RATE && network->get_output_mode() != Nodes::Network::OutputMode::GRAPHICS_BIND) {
62 MF_WARN(Journal::Component::API,
63 Journal::Context::Init,
64 "Registering visual network in VISUAL_RATE domain without GRAPHICS_BIND output mode. Forcing GRAPHICS_BIND mode.");
65 network->set_output_mode(Nodes::Network::OutputMode::GRAPHICS_BIND);
66 }
67
68 register_node_network(network, token);
69}

71void register_buffer(const std::shared_ptr<Buffers::Buffer>& buffer, const CreationContext& ctx)
72{
73 auto token = get_buffer_token(ctx.domain.value());
74
75 if (auto audio_buffer = std::dynamic_pointer_cast<Buffers::AudioBuffer>(buffer)) {
76 if (ctx.channel.has_value()) {
77 register_audio_buffer(audio_buffer, ctx.channel.value());
78 } else if (ctx.channels.has_value()) {
79 clone_buffer_to_channels(audio_buffer, ctx.channels.value(), token);
80 }
81 return;
82 }
83
84 if (auto vk_buffer = std::dynamic_pointer_cast<Buffers::VKBuffer>(buffer)) {
85 register_graphics_buffer(vk_buffer, token);
86 return;
87 }

157{
