12 Core::NetworkTransport resolve_transport(NetworkTransportHint hint, StreamProfile profile)
13 {
14 switch (hint) {
15 case NetworkTransportHint::UDP:
16 return Core::NetworkTransport::UDP;
17 case NetworkTransportHint::TCP:
18 return Core::NetworkTransport::TCP;
19 case NetworkTransportHint::SHARED_MEMORY:
20 return Core::NetworkTransport::SHARED_MEMORY;
21 case NetworkTransportHint::AUTO:
22 switch (profile) {
23 case StreamProfile::ORDERED_BULK:
24 return Core::NetworkTransport::TCP;
