133void GraphicsSubsystem::process_frame_coroutines_impl(
134 const std::vector<std::shared_ptr<Vruta::Routine>>& tasks,
135 uint64_t processing_units)
136{
137 if (tasks.empty()) {
138 return;
139 }
140
141 uint64_t current_frame = m_frame_clock->current_position();
142
143 if (processing_units == 0) {
144 processing_units = 1;
145 }
146
147 for (uint64_t i = 0; i < processing_units; ++i) {
148 uint64_t frame_to_process = current_frame + i;
149
150 for (auto& routine : tasks) {
151 if (!routine || !routine->is_active()) {
152 continue;
153 }
154
155 if (routine->requires_clock_sync()) {
156 if (frame_to_process >= routine->next_execution()) {
157 routine->try_resume(frame_to_process);
158 }
159 } else {
160 routine->try_resume(frame_to_process);
161 }
162 }
163 }

166void GraphicsSubsystem::start()
167{

228 "Graphics Subsystem stopped.");

277void GraphicsSubsystem::process()
278{
279 if (!m_handle) {
280 return;
281 }
282
283 for (auto& [name, hook] : m_handle->pre_process_hooks) {
284 hook(1);
285 }
286
287 m_handle->tasks.process(1);
288
289 m_handle->nodes.process(1);
290
291 m_handle->buffers.process(1);
292
293 register_windows_for_processing();
294 m_backend->handle_window_resize();
295 render_all_windows();
296
297 m_handle->windows.process();
298
299 cleanup_closed_windows();
300 for (auto& [name, hook] : m_handle->post_process_hooks) {
301 hook(1);
302 }

357void GraphicsSubsystem::graphics_thread_loop()
358{
359 while (m_running.load(std::memory_order_acquire)) {
360 if (m_paused.load(std::memory_order_acquire)) {
361 std::this_thread::sleep_for(std::chrono::milliseconds(16));
362 continue;
363 }
364
365 m_frame_clock->tick();
366
367 process();
368
369 m_frame_clock->wait_for_next_frame();
370
371 if (m_frame_clock->is_frame_late()) {
372 uint64_t lag = m_frame_clock->get_frame_lag();
373 if (lag > 2) {
374 MF_RT_WARN(Journal::Component::Core, Journal::Context::GraphicsSubsystem,
375 "Frame lag detected: {} frames behind (Measured FPS: {:.1f})",
376 lag, m_frame_clock->get_measured_fps());
377 }
378 }
379 }
