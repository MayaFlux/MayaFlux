56InspectResult Inspector::scheduler(

59 LayoutCursor& cursor,

73 layer, context, window, cursor,

75
76 InspectResult result;
77 result.group = std::move(root_group);

84 auto task_result = inspect_task(

86 x_min, x_max, row_h);
87
88 result.group.header.attach(layer, task_result.group.header.header_id);
89 result.children.push_back(std::move(task_result));
90 }
91
92 return result;
