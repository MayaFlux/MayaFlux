41/**
42 * @brief Derive a GpuBufferBinding list from a ShaderSpec.
43 *
44 * Translates ShaderSpec binding declarations to the GpuBufferBinding
45 * vocabulary understood by GpuResourceManager and any GpuExecutionContext
46 * subclass. The result is suitable for passing to declare_buffer_bindings()
47 * overrides or directly to GpuResourceManager::initialise().
48 */
49[[nodiscard]] inline std::vector<GpuBufferBinding> bindings_from_spec(
50 const Portal::Graphics::ShaderSpec& spec)
51{
52 std::vector<GpuBufferBinding> out;
53 out.reserve(spec.bindings.size());
54 for (const auto& b : spec.bindings) {
55 out.push_back({
56 .set = 0,
57 .binding = b.binding_index,
58 .direction = to_binding_direction(b.direction),
59 .element_type = to_element_type(b.modality),
60 });
61 }
62 return out;
63}
