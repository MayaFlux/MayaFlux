21/**
22 * @struct MappedState
23 * @brief Value carrier for a Mapped primitive.
24 *
25 * Owns the current value and a version counter. Any writer increments
26 * version after updating value. The graphics tick reads both and decides
27 * whether to regenerate geometry.
28 *
29 * Exposed as shared_ptr<MappedState<T>> so any external system — a node
30 * output reader, a buffer writer, a scheduler callback — can hold a
31 * reference and read or write without knowing about the primitive.
32 *
33 * @tparam T Value type. float for a single continuous parameter,
34 * glm::vec2 for a 2D position, any plain data type.
35 */
36template <typename T>
37struct MappedState {
38 T value {};
39 Version version { 0 };
40 uint32_t id { 0 };
41
42 void write(T v)
43 {
44 value = v;
45 ++version;
46 }
47};

49/**
50 * @brief Geometry function signature.
51 *
52 * Called by the graphics tick when the version has advanced. Receives the
53 * current value and an output byte vector to fill with interleaved vertex
54 * data. Layout, stride, and topology are determined by how the caller
55 * constructed the VKBuffer and RenderProcessor — the function is agnostic
56 * to all of those.
57 *
58 * The function is also responsible for updating the Element's spatial
59 * description if the geometry change affects hit testing (e.g. a handle
60 * that moves). It receives the Element by reference for that purpose.
61 *
62 * @tparam T Value type matching the MappedState.
63 */
64template <typename T>
65using GeometryFn = std::function<void(T value, std::vector<uint8_t>& out_bytes, Element& element)>;

67/**
68 * @struct Mapped
69 * @brief Infrastructure for a continuously-mapped value whose GPU geometry
70 * tracks it.
71 *
72 * The geometry function IS the primitive. Mapped provides only the
73 * mechanism: a shared value carrier, a version counter, and a sync()
74 * call that invokes the geometry function when the value changes.
75 *
76 * The geometry function decides everything about shape, topology, and
77 * vertex layout. It can write from any source — mouse pixel coordinates,
78 * microphone energy, another node's output, a tendency field, a computed
79 * function of time — by whatever means the caller chooses. The common
80 * geometry helpers in Geometry.hpp are illustrative, not idiomatic.
81 *
82 * Orchestration (node binding, buffer mapping, scheduler-driven update,
83 * input capture) attaches externally by writing to state and calling
84 * state->write(v). Mapped has no knowledge of or dependency on any
85 * orchestration layer.
86 *
87 * @tparam T Value type.
88 */
89template <typename T>
90struct Mapped {
91 /// @brief Shared value carrier. External systems hold a copy of this ptr.
92 std::shared_ptr<MappedState<T>> state;
93
94 /// @brief Geometry regeneration function. Set once at construction.
95 GeometryFn<T> geometry_fn;
96
97 /// @brief The Element registered with the Layer.
98 /// element.id is valid after Layer::add(element) has been called.
99 Element element;
100
101 /// @brief Last version seen by sync(). Initialised to a sentinel that
102 /// guarantees one geometry generation on the first sync() call.
103 Version last_version { static_cast<Version>(-1) };
104
105 /**
106 * @brief Call once per graphics tick.
107 *
108 * If state->version has advanced since last_version, invokes geometry_fn
109 * with the current value into a local byte buffer, then passes the result
110 * to VKBuffer::set_data so the existing upload path handles GPU transfer.
111 *
112 * The caller is responsible for driving the graphics tick — Mapped does
113 * not register itself with any scheduler.
114 */
115 void sync()
116 {
117 if (!state || !geometry_fn || !element.buffer)
118 return;
119 if (state->version == last_version)
120 return;
121
122 geometry_fn(state->value, m_bytes, element);
123 element.buffer->submit(m_bytes);
124 last_version = state->version;
125 }
126
127private:
128 std::vector<uint8_t> m_bytes;
129};

131/**
132 * @brief Construct a Mapped<T> with a freshly allocated MappedState.
133 * @param initial Starting value.
134 * @param geometry Geometry function producing vertex bytes from a value.
135 * @param buffer VKBuffer whose raw bytes the geometry function writes into.
136 * @return Ready-to-use Mapped. Register element with a Layer before first sync().
137 */
138template <typename T>
139[[nodiscard]] Mapped<T> make_mapped(
140 T initial,
141 GeometryFn<T> geometry,
142 std::shared_ptr<Buffers::FormaBuffer> buffer)
143{
144 auto st = std::make_shared<MappedState<T>>();
145 st->write(initial);
146
147 Element el;
148 el.buffer = std::move(buffer);
149
150 Mapped<T> m;
151 m.state = std::move(st);
152 m.geometry_fn = std::move(geometry);
153 m.element = std::move(el);
154 m.last_version = static_cast<Version>(-1);
155 return m;
156}
