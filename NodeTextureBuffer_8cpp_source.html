92void NodeTextureBuffer::setup_rendering(const RenderConfig& user_config)
93{
94 if (!user_config.vertex_shader.empty()) {
95 m_render_config.vertex_shader = user_config.vertex_shader;
96 }
97 if (!user_config.fragment_shader.empty()) {
98 m_render_config.fragment_shader = user_config.fragment_shader;
99 }
100 if (!user_config.default_texture_binding.empty()) {
101 m_render_config.default_texture_binding = user_config.default_texture_binding;
102 }
103 if (user_config.topology != Portal::Graphics::PrimitiveTopology::TRIANGLE_STRIP) {
104 MF_WARN(Journal::Component::Buffers, Journal::Context::BufferProcessing,
105 "NodeTextureBuffer currently only supports TRIANGLE_STRIP topology. Ignoring provided topology.");
106 } else {
107 if (user_config.topology != m_render_config.topology) {
108 m_render_config.topology = user_config.topology;
109 }
110 }
111 m_render_config.target_window = user_config.target_window;
112
113 if (!user_config.additional_textures.empty()) {
114 for (const auto& [name, texture] : user_config.additional_textures) {
115 m_render_config.additional_textures.emplace_back(name, texture);
116 }
117 }
118
119 if (!m_render_processor) {
120 ShaderConfig shader_config { m_render_config.vertex_shader };
121 shader_config.bindings[m_render_config.default_texture_binding] = ShaderBinding(
122 0, 0, vk::DescriptorType::eCombinedImageSampler);
123
124 uint32_t binding_index = 1;
125 for (const auto& [name, _] : m_render_config.additional_textures) {
126 shader_config.bindings[name] = ShaderBinding(
127 0, binding_index++, vk::DescriptorType::eCombinedImageSampler);
128 }
129
130 m_render_processor = std::make_shared<RenderProcessor>(shader_config);
131 }
132
133 m_render_processor->set_fragment_shader(m_render_config.fragment_shader);
134 m_render_processor->set_target_window(m_render_config.target_window, std::dynamic_pointer_cast<VKBuffer>(shared_from_this()));
135 m_render_processor->set_primitive_topology(m_render_config.topology);
136
137 m_render_processor->bind_texture(m_render_config.default_texture_binding, get_gpu_texture());
138
139 for (const auto& [name, texture] : m_render_config.additional_textures) {
140 m_render_processor->bind_texture(name, texture);
141 }
142
143 get_processing_chain()->add_final_processor(m_render_processor, shared_from_this());
144
145 MF_INFO(Journal::Component::Buffers, Journal::Context::Init,
146 "NodeTextureBuffer '{}' rendering configured: shader={}, topology={}",
147 m_binding_name, m_render_config.fragment_shader,
148 static_cast<int>(m_render_config.topology));
149}

172void NodeTextureBuffer::generate_fullscreen_quad()
173{
174 struct QuadVertex {
175 glm::vec3 position;
176 glm::vec2 texcoord;
177 };
178
179 const std::vector<QuadVertex> quad = {
180 { { -1.0F, -1.0F, 0.0F }, { 0.0F, 1.0F } },
181 { { 1.0F, -1.0F, 0.0F }, { 1.0F, 1.0F } },
182 { { -1.0F, 1.0F, 0.0F }, { 0.0F, 0.0F } },
183 { { 1.0F, 1.0F, 0.0F }, { 1.0F, 0.0F } }
184 };
185
186 m_vertex_bytes.resize(quad.size() * sizeof(QuadVertex));
187 std::memcpy(m_vertex_bytes.data(), quad.data(), m_vertex_bytes.size());
188
189 // Set vertex layout
190 Kakshya::VertexLayout vertex_layout {
191 .vertex_count = 4,
192 .stride_bytes = sizeof(QuadVertex),
193 .attributes = {
194 { .component_modality = Kakshya::DataModality::VERTEX_POSITIONS_3D,
195 .offset_in_vertex = offsetof(QuadVertex, position),
196 .name = "position" },
197 { .component_modality = Kakshya::DataModality::TEXTURE_COORDS_2D,
198 .offset_in_vertex = offsetof(QuadVertex, texcoord),
199 .name = "texcoord" } }
200 };
201
202 set_vertex_layout(vertex_layout);
203
204 MF_DEBUG(Journal::Component::Buffers, Journal::Context::BufferProcessing,
205 "NodeTextureBuffer: generated fullscreen quad");
206}
