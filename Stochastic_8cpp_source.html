55double Random::generate_distributed_sample()
56{
57 switch (m_type) {
58 case Utils::distribution::UNIFORM: {
59 std::uniform_real_distribution<double> dist(m_current_start, m_current_end);
60 return dist(m_random_engine);
61 }
62 case Utils::distribution::NORMAL: {
63 const double range = m_current_end - m_current_start;
64 std::normal_distribution<double> dist(0.0, range / m_normal_spread);
65 return dist(m_random_engine);
66 }
67 case Utils::distribution::EXPONENTIAL: {
68 std::exponential_distribution<double> dist(1.0);
69 return dist(m_random_engine);
70 }
71 case Utils::distribution::POISSON: {
72 std::poisson_distribution<int> dist(m_current_end - m_current_start);
73 return static_cast<double>(dist(m_random_engine));
74 }
75 default:
76 throw std::invalid_argument("Invalid distribution type");
77 }
78}

80double Random::transform_sample(double sample, double start, double end) const
81{
82 if (m_type == Utils::distribution::NORMAL) {
83 sample = std::max(start, std::min(end, sample));
84 } else if (m_type == Utils::distribution::EXPONENTIAL) {
85 sample /= end;
