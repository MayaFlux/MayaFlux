82double Impulse::process_sample(double input)
83{
84 double output = 0.0;
85 m_impulse_occurred = false;
86
87 double effective_freq = m_frequency;
88 if (m_frequency_modulator) {
89 atomic_inc_modulator_count(m_frequency_modulator->m_modulator_count, 1);
90 uint32_t state = m_frequency_modulator->m_state.load();
91 if (state & Utils::NodeState::PROCESSED) {
92 effective_freq += m_frequency_modulator->get_last_output();
93 } else {
94 effective_freq = m_frequency_modulator->process_sample(0.F);
95 atomic_add_flag(m_frequency_modulator->m_state, Utils::NodeState::PROCESSED);
96 }
97
98 if (effective_freq <= 0) {
99 effective_freq = 0.001;
100 }
101 update_phase_increment(effective_freq);
102 }
103
104 if (m_phase < m_phase_inc) {
105 output = m_amplitude;
106 m_impulse_occurred = true;
107 } else {
108 output = 0.0;
109 }
110
111 double current_amplitude = m_amplitude;
112 if (m_amplitude_modulator) {
113 atomic_inc_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
114 uint32_t state = m_amplitude_modulator->m_state.load();
115
116 if (state & Utils::NodeState::PROCESSED) {
117 current_amplitude += m_amplitude_modulator->get_last_output();
118 } else {
119 current_amplitude += m_amplitude_modulator->process_sample(0.F);
120 atomic_add_flag(m_amplitude_modulator->m_state, Utils::NodeState::PROCESSED);
121 }
122 }
123
124 output *= current_amplitude;
125 m_amplitude = current_amplitude;
126 output += m_offset;
127
128 m_phase += m_phase_inc;
129 if (m_phase >= 1.0) {
130 m_phase -= 1.0;
131 }
132
133 m_last_output = output;
134
135 if ((!m_state_saved || (m_state_saved && m_fire_events_during_snapshot))
136 && !m_networked_node) {
137 notify_tick(output);
138 }
139
140 if (m_frequency_modulator) {
141 atomic_dec_modulator_count(m_frequency_modulator->m_modulator_count, 1);
142 try_reset_processed_state(m_frequency_modulator);
143 }
144 if (m_amplitude_modulator) {
145 atomic_dec_modulator_count(m_amplitude_modulator->m_modulator_count, 1);
146 try_reset_processed_state(m_amplitude_modulator);
147 }
148
149 return output;
150}

184void Impulse::notify_tick(double value)
185{
186 update_context(value);
187 auto& ctx = get_last_context();
188
189 for (auto& callback : m_callbacks) {
190 callback(ctx);
191 }
192 for (auto& [callback, condition] : m_conditional_callbacks) {
193 if (condition(ctx)) {
194 callback(ctx);
195 }
196 }
197 if (m_impulse_occurred) {
198 for (auto& callback : m_impulse_callbacks) {
199 callback(ctx);
200 }
