15/**
16 * @class TextureExecutionContext
17 * @brief GpuExecutionContext specialisation for image compute shaders.
18 *
19 * Accepts DataIO (Datum<vector<DataVariant>>) as both input and output,
20 * matching the standard ComputeOperation contract. Image staging is driven
21 * by the optional container field on the input Datum:
22 *
23 * - When datum.container holds a TextureContainer, extract_inputs() stashes
24 * it and on_before_gpu_dispatch() uploads it to a VKImage staged at the
25 * declared image input binding. This is the preferred path: the container
26 * carries format, layer count, mip metadata, and sampler config.
27 *
28 * - When datum.container is absent or not a TextureContainer, no image
29 * staging occurs. The shader receives whatever bindings the caller
30 * declared via set_binding_data / stage_passthrough / stage_image_sampled
31 * directly. dispatch_async still works; it is a no-op on the image side.
32 *
33 * Output options (select one at construction):
34 *
35 * OutputMode::CONTAINER — the storage image at binding 0 is downloaded into
36 * a new TextureContainer and placed in the output Datum's container field.
37 * Primary float readback is empty. Mirrors the original behaviour for
38 * texture-transform pipelines (blur, colour grade, format convert).
39 *
40 * OutputMode::SCALAR — no image download. collect_result() returns the
41 * primary float readback and aux SSBOs from GpuChannelResult, exactly as
42 * ShaderExecutionContext does. Use for reduction shaders that write a small
43 * SSBO (spatial average, histogram, region sample) and feed the node graph.
44 *
45 * dispatch_async / collect_result mirror ShaderExecutionContext so
46 * GpuComputeNode can own this class without modification.
47 *
48 * calculate_dispatch_size() uses image dimensions when a TextureContainer is
49 * present, otherwise falls through to the standard element-count path.
50 */

110 *
111 * If datum.container holds a TextureContainer it is uploaded and staged

113 * Call collect_result() (SCALAR mode) or retrieve the output Datum via
114 * execute() (CONTAINER mode) once signaled.

117 * @return FenceID to poll with ShaderFoundry::is_fence_signaled.

246 /**
247 * @brief Stages the pending TextureContainer as IMAGE_SAMPLED and,
248 * in CONTAINER mode, allocates the output storage image at binding 0.
249 */
250 void on_before_gpu_dispatch(
251 const std::vector<std::vector<double>>& /*channels*/,
252 const DataStructureInfo& /*structure_info*/) override
253 {
254 if (m_pending_container) {
255 const auto sampler = Portal::Graphics::SamplerForge::instance().get_default_linear();
256 std::shared_ptr<Core::VKImage> img;
257 if (m_pending_container->get_layer_count() > 1) {
258 if (!m_upload_staging) {
259 m_upload_staging = Buffers::create_image_staging_buffer(
260 m_pending_container->byte_size() * m_pending_container->get_layer_count());
261 }
262 img = m_pending_container->to_image_array(m_upload_staging);
263 } else {
264 if (!m_upload_staging) {
265 m_upload_staging = Buffers::create_image_staging_buffer(
266 m_pending_container->byte_size());
267 }
268 img = m_pending_container->to_image(m_pending_layer, m_upload_staging);
269 }
270 stage_image_sampled(m_image_binding, std::move(img), sampler);
271
272 if (m_output_mode == OutputMode::CONTAINER
273 || m_output_mode == OutputMode::IMAGE) {
274 prepare_output_image(m_pending_container->get_width(),
275 m_pending_container->get_height());
276 }
277 }

280 /**
281 * @brief Derives dispatch size from the staged container dimensions when
282 * available; falls back to element-count dispatch otherwise.
283 */
284 [[nodiscard]] std::array<uint32_t, 3> calculate_dispatch_size(
285 size_t total_elements,
286 const DataStructureInfo& structure_info) const override
287 {
288 if (m_pending_container) {
289 const auto& ws = gpu_config().workgroup_size;
290 const uint32_t w = m_pending_container->get_width();
291 const uint32_t h = m_pending_container->get_height();
292 return {
293 (w + ws[0] - 1) / ws[0],
294 (h + ws[1] - 1) / ws[1],
295 1U
296 };
297 }
298 return Base::calculate_dispatch_size(total_elements, structure_info);
299 }

301 /**
302 * @brief In CONTAINER mode: downloads the storage image at binding 0 into
303 * a new TextureContainer placed in output.container.
304 * In SCALAR mode: returns an empty DataIO (use collect_result() instead).
305 */
306 output_type collect_gpu_outputs(
307 const GpuChannelResult& /*raw*/,
308 const std::vector<std::vector<double>>& /*channels*/,
309 const DataStructureInfo& /*structure_info*/) override
310 {
311 if (m_output_mode == OutputMode::SCALAR)
312 return output_type {};
313
314 auto img = get_output_image(0);
315 if (!img) {
316 error<std::runtime_error>(
317 Journal::Component::Yantra,
318 Journal::Context::BufferProcessing,
319 std::source_location::current(),
320 "TextureExecutionContext: no output image at binding 0 after dispatch");
321 }
322
323 Portal::Graphics::TextureLoom::instance().transition_layout(
324 img, vk::ImageLayout::eGeneral, vk::ImageLayout::eShaderReadOnlyOptimal);
325
326 if (m_output_mode == OutputMode::IMAGE)
327 return output_type {};
328
329 const uint32_t w = m_pending_container ? m_pending_container->get_width() : img->get_width();
330 const uint32_t h = m_pending_container ? m_pending_container->get_height() : img->get_height();
331
332 if (!m_output_container) {
333 m_output_container = std::make_shared<Kakshya::TextureContainer>(
334 w, h, m_output_format);
335 }
336
337 if (!m_download_staging) {
338 m_download_staging = Buffers::create_image_staging_buffer(
339 m_output_container->byte_size());
340 }
341
342 m_output_container->from_image(img, m_download_staging, 0);
343
344 output_type result;
345 result.container = std::static_pointer_cast<Kakshya::SignalSourceContainer>(
346 m_output_container);
347 return result;
348 }
