59{
60 if (!window) {

74 if (!glfw_handle) {

81 VkResult result = Parallel::dispatch_main_sync([&]() {
82 return glfwCreateWindowSurface(

84 glfw_handle,

86 &c_surface);

91 "Failed to create window surface");
92 return nullptr;

99 "Surface created for window '{}'", window->get_create_info().title);
