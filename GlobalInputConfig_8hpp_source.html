13/**
14 * @brief Filter for HID device enumeration
15 *
16 * Used to selectively enumerate HID devices by VID/PID or usage page/usage.
17 * All fields are optional - nullopt means "match any".
18 */
19struct MAYAFLUX_API HIDDeviceFilter {
20 std::optional<uint16_t> vendor_id; ///< USB Vendor ID (nullopt = any)
21 std::optional<uint16_t> product_id; ///< USB Product ID (nullopt = any)
22 std::optional<uint16_t> usage_page; ///< HID usage page (nullopt = any)
23 std::optional<uint16_t> usage; ///< HID usage (nullopt = any)
24
25 /**
26 * @brief Check if a device matches this filter
27 */
28 [[nodiscard]] bool matches(uint16_t vid, uint16_t pid,
29 uint16_t upage = 0, uint16_t usg = 0) const
30 {
31 if (vendor_id && *vendor_id != vid)
32 return false;
33 if (product_id && *product_id != pid)
34 return false;
35 if (usage_page && *usage_page != upage)
36 return false;
37
38 return !usage || *usage == usg;
39 }

245/**
246 * @class GlobalInputConfig
247 * @brief Configuration for the InputSubsystem
248 *
249 * Centralizes configuration for all input backends (HID, MIDI, OSC, Serial).
250 * Passed to InputSubsystem during construction.
251 *
252 * Example usage:
253 * @code
254 * GlobalInputConfig input_config;
255 *
256 * // Enable HID with gamepad filter
257 * input_config.hid.enabled = true;
258 * input_config.hid.filters.push_back(HIDDeviceFilter::controller());
259 *
260 * // Enable OSC on port 8000
261 * input_config.osc.enabled = true;
262 * input_config.osc.receive_port = 8000;
263 *
264 * auto input_subsystem = std::make_unique<InputSubsystem>(input_config);
265 * @endcode
266 */
267struct MAYAFLUX_API GlobalInputConfig {
268 HIDBackendInfo hid; ///< HID backend configuration
269 MIDIBackendInfo midi; ///< MIDI backend configuration
270 OSCConfigInfo osc; ///< OSC backend configuration
271 SerialBackendInfo serial; ///< Serial backend configuration
272
